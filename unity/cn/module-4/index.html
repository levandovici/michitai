模块4：Unity动画和Animator

Unity动画系统和Animator控制的综合示例

基础Animator控制

控制Animator参数并从脚本触发动画。

using UnityEngine; public class AnimatorController : MonoBehaviour { private Animator animator; void Start() { animator = GetComponent<Animator>(); Debug.Log("为 " + gameObject.name + " 初始化Animator"); } void Update() { if (Input.GetKeyDown(KeyCode.W)) { animator.SetBool("IsWalking", true); Debug.Log("激活行走动画"); } if (Input.GetKeyUp(KeyCode.W)) { animator.SetBool("IsWalking", false); Debug.Log("停止行走动画"); } if (Input.GetKeyDown(KeyCode.Space)) { animator.SetTrigger("Jump"); Debug.Log("触发跳跃动画"); } } public void SetAnimationSpeed(float speed) { animator.SetFloat("Speed", speed); Debug.Log("动画速度设置为: " + speed); } }

动画状态管理

管理状态之间的转换并监控动画进度。

using UnityEngine; public class AnimationStateManager : MonoBehaviour { private Animator animator; private Rigidbody rb; [Header("动画参数")] public float walkSpeed = 2f; public float runSpeed = 5f; private bool isGrounded = true; void Start() { animator = GetComponent<Animator>(); rb = GetComponent<Rigidbody>(); Debug.Log("动画状态管理器已初始化"); } void Update() { HandleMovementInput(); UpdateAnimationParameters(); MonitorAnimationStates(); } void HandleMovementInput() { float horizontal = Input.GetAxis("Horizontal"); float vertical = Input.GetAxis("Vertical"); Vector3 movement = new Vector3(horizontal, 0, vertical); bool isMoving = movement.magnitude > 0.1f; bool isRunning = Input.GetKey(KeyCode.LeftShift); animator.SetBool("IsWalking", isMoving && !isRunning); animator.SetBool("IsRunning", isMoving && isRunning); animator.SetFloat("MovementSpeed", movement.magnitude); } void UpdateAnimationParameters() { animator.SetBool("IsGrounded", isGrounded); animator.SetFloat("VerticalVelocity", rb.velocity.y); } void MonitorAnimationStates() { AnimatorStateInfo currentState = animator.GetCurrentAnimatorStateInfo(0); if (currentState.IsName("Jump") && currentState.normalizedTime >= 1.0f) { Debug.Log("跳跃动画完成"); } } }

混合树控制

使用混合树实现流畅自然的动画。

using UnityEngine; public class BlendTreeController : MonoBehaviour { private Animator animator; [Header("混合树设置")] public float accelerationTime = 2f; public float decelerationTime = 1f; private float currentBlendX = 0f; private float currentBlendY = 0f; private float targetBlendX = 0f; private float targetBlendY = 0f; void Start() { animator = GetComponent<Animator>(); Debug.Log("混合树控制器已初始化"); } void Update() { HandleInput(); UpdateBlendValues(); SetAnimatorParameters(); } void HandleInput() { float inputX = Input.GetAxis("Horizontal"); float inputY = Input.GetAxis("Vertical"); targetBlendX = inputX; targetBlendY = inputY; if (Input.GetKey(KeyCode.LeftShift)) { targetBlendX *= 2f; targetBlendY *= 2f; } targetBlendX = Mathf.Clamp(targetBlendX, -1f, 1f); targetBlendY = Mathf.Clamp(targetBlendY, -1f, 1f); } void UpdateBlendValues() { float rate = 1f / accelerationTime; currentBlendX = Mathf.MoveTowards(currentBlendX, targetBlendX, rate * Time.deltaTime); currentBlendY = Mathf.MoveTowards(currentBlendY, targetBlendY, rate * Time.deltaTime); } void SetAnimatorParameters() { animator.SetFloat("BlendX", currentBlendX); animator.SetFloat("BlendY", currentBlendY); float magnitude = new Vector2(currentBlendX, currentBlendY).magnitude; animator.SetFloat("Speed", magnitude); animator.SetBool("IsMoving", magnitude > 0.1f); } }

动画事件

使用动画事件将效果与动画同步。

using UnityEngine; public class AnimationEventHandler : MonoBehaviour { [Header("音频组件")] public AudioSource audioSource; public AudioClip footstepSound; public AudioClip jumpSound; [Header("视觉效果")] public ParticleSystem dustEffect; void Start() { if (audioSource == null) audioSource = GetComponent<AudioSource>(); Debug.Log("动画事件处理器已初始化"); } // 由动画事件调用的脚步方法 public void OnFootstep() { Debug.Log("检测到脚步事件"); if (audioSource && footstepSound) { audioSource.PlayOneShot(footstepSound); } if (dustEffect) { dustEffect.Play(); } } // 跳跃开始方法 public void OnJumpStart() { Debug.Log("跳跃开始"); if (audioSource && jumpSound) { audioSource.PlayOneShot(jumpSound); } } // 着陆方法 public void OnLanding() { Debug.Log("通过事件检测到着陆"); if (dustEffect) { dustEffect.Play(); } CameraShake(); } // 攻击事件方法 public void OnAttackHit() { Debug.Log("攻击命中时刻"); Collider[] enemies = Physics.OverlapSphere(transform.position, 2f); foreach (Collider enemy in enemies) { if (enemy.CompareTag("Enemy")) { Debug.Log("击中敌人: " + enemy.name); } } } void CameraShake() { Camera mainCamera = Camera.main; if (mainCamera) { StartCoroutine(ShakeCoroutine(mainCamera)); } } System.Collections.IEnumerator ShakeCoroutine(Camera camera) { Vector3 originalPosition = camera.transform.position; float duration = 0.2f; float magnitude = 0.1f; float elapsed = 0f; while (elapsed < duration) { float x = Random.Range(-1f, 1f) * magnitude; float y = Random.Range(-1f, 1f) * magnitude; camera.transform.position = originalPosition + new Vector3(x, y, 0); elapsed += Time.deltaTime; yield return null; } camera.transform.position = originalPosition; } }

高级动画脚本

高级动画技术，包括IK和程序化控制。

using UnityEngine; public class AdvancedAnimationController : MonoBehaviour { private Animator animator; [Header("IK设置")] public Transform leftHandTarget; public Transform rightHandTarget; public Transform lookAtTarget; [Range(0f, 1f)] public float leftHandWeight = 1f; [Range(0f, 1f)] public float rightHandWeight = 1f; [Range(0f, 1f)] public float lookAtWeight = 1f; void Start() { animator = GetComponent<Animator>(); Debug.Log("高级动画控制器已初始化"); } void Update() { HandleAdvancedControls(); } void HandleAdvancedControls() { if (Input.GetKey(KeyCode.Plus)) { animator.speed = Mathf.Min(animator.speed + Time.deltaTime, 3f); Debug.Log("动画速度: " + animator.speed); } if (Input.GetKey(KeyCode.Minus)) { animator.speed = Mathf.Max(animator.speed - Time.deltaTime, 0.1f); Debug.Log("动画速度: " + animator.speed); } if (Input.GetKeyDown(KeyCode.R)) { animator.speed = 1f; Debug.Log("动画速度已重置"); } } // IK回调 - 由Unity自动调用 void OnAnimatorIK(int layerIndex) { if (animator) { // 设置左手的位置和旋转 if (leftHandTarget) { animator.SetIKPositionWeight(AvatarIKGoal.LeftHand, leftHandWeight); animator.SetIKRotationWeight(AvatarIKGoal.LeftHand, leftHandWeight); animator.SetIKPosition(AvatarIKGoal.LeftHand, leftHandTarget.position); animator.SetIKRotation(AvatarIKGoal.LeftHand, leftHandTarget.rotation); } // 设置右手的位置和旋转 if (rightHandTarget) { animator.SetIKPositionWeight(AvatarIKGoal.RightHand, rightHandWeight); animator.SetIKRotationWeight(AvatarIKGoal.RightHand, rightHandWeight); animator.SetIKPosition(AvatarIKGoal.RightHand, rightHandTarget.position); animator.SetIKRotation(AvatarIKGoal.RightHand, rightHandTarget.rotation); } // 设置注视目标 if (lookAtTarget) { animator.SetLookAtWeight(lookAtWeight); animator.SetLookAtPosition(lookAtTarget.position); } } } public void SetLayerWeight(int layerIndex, float weight) { animator.SetLayerWeight(layerIndex, weight); Debug.Log($"层 {layerIndex} 权重设置为 {weight}"); } public void PlayAnimationAtTime(string stateName, float normalizedTime) { animator.Play(stateName, 0, normalizedTime); Debug.Log($"动画 {stateName} 在时间 {normalizedTime} 播放"); } }
