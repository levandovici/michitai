模块7：多线程编程

Unity中多线程和并行编程的全面示例

线程基础知识

线程介绍和Thread类的基本操作。

using System; using System.Threading; using UnityEngine; public class ThreadingBasics : MonoBehaviour { [Header("线程设置")] public int numberOfThreads = 4; public bool useBackgroundThreads = true; private Thread[] workerThreads; private bool isRunning = false; void Start() { InitializeThreads(); Debug.Log("线程系统已初始化"); } void InitializeThreads() { workerThreads = new Thread[numberOfThreads]; for (int i = 0; i < numberOfThreads; i++) { int threadId = i; workerThreads[i] = new Thread(() => WorkerThreadMethod(threadId)); workerThreads[i].IsBackground = useBackgroundThreads; workerThreads[i].Name = $"工作线程 {threadId}"; } } void WorkerThreadMethod(int threadId) { Debug.Log($"线程 {threadId} 在核心 {Thread.CurrentThread.ManagedThreadId} 上启动"); while (isRunning) { PerformWork(threadId); Thread.Sleep(1000); } Debug.Log($"线程 {threadId} 已完成"); } void PerformWork(int threadId) { int result = 0; for (int i = 0; i < 1000000; i++) { result += i % (threadId + 1); } Debug.Log($"线程 {threadId} 完成工作。结果: {result}"); } void Update() { if (Input.GetKeyDown(KeyCode.S)) { StartThreads(); } if (Input.GetKeyDown(KeyCode.T)) { StopThreads(); } } public void StartThreads() { if (!isRunning) { isRunning = true; for (int i = 0; i < workerThreads.Length; i++) { if (workerThreads[i].ThreadState == ThreadState.Unstarted) { workerThreads[i].Start(); } } Debug.Log("所有线程已启动"); } } public void StopThreads() { if (isRunning) { isRunning = false; Debug.Log("正在停止线程..."); } } void OnDestroy() { StopThreads(); } }

任务并行库

使用现代API进行并行编程。

using System; using System.Collections.Generic; using System.Threading.Tasks; using UnityEngine; public class TaskParallelController : MonoBehaviour { [Header("任务设置")] public int dataSize = 1000000; private List<int> largeDataSet; private CancellationTokenSource cancellationTokenSource; void Start() { InitializeData(); Debug.Log("任务并行库控制器已初始化"); } void InitializeData() { largeDataSet = new List<int>(dataSize); for (int i = 0; i < dataSize; i++) { largeDataSet.Add(UnityEngine.Random.Range(1, 1000)); } Debug.Log($"已创建包含 {dataSize} 个元素的数据集"); } void Update() { if (Input.GetKeyDown(KeyCode.P)) { ProcessDataParallel(); } if (Input.GetKeyDown(KeyCode.S)) { ProcessDataSequential(); } if (Input.GetKeyDown(KeyCode.A)) { ProcessDataAsync(); } } public void ProcessDataParallel() { Debug.Log("开始并行处理..."); var stopwatch = System.Diagnostics.Stopwatch.StartNew(); long sum = 0; object lockObject = new object(); Parallel.ForEach(largeDataSet, (number) => { long result = ComputeExpensiveOperation(number); lock (lockObject) { sum += result; } }); stopwatch.Stop(); Debug.Log($"并行处理在 {stopwatch.ElapsedMilliseconds} 毫秒内完成。总和: {sum}"); } public void ProcessDataSequential() { Debug.Log("开始顺序处理..."); var stopwatch = System.Diagnostics.Stopwatch.StartNew(); long sum = 0; foreach (int number in largeDataSet) { sum += ComputeExpensiveOperation(number); } stopwatch.Stop(); Debug.Log($"顺序处理在 {stopwatch.ElapsedMilliseconds} 毫秒内完成。总和: {sum}"); } public async void ProcessDataAsync() { Debug.Log("开始异步处理..."); cancellationTokenSource = new CancellationTokenSource(); try { var result = await ProcessDataAsyncInternal(cancellationTokenSource.Token); Debug.Log($"异步处理完成。结果: {result}"); } catch (OperationCanceledException) { Debug.Log("异步操作已取消"); } } private async Task<long> ProcessDataAsyncInternal(CancellationToken cancellationToken) { return await Task.Run(() => { long sum = 0; for (int i = 0; i < largeDataSet.Count; i++) { cancellationToken.ThrowIfCancellationRequested(); sum += ComputeExpensiveOperation(largeDataSet[i]); } return sum; }, cancellationToken); } private long ComputeExpensiveOperation(int input) { long result = 0; for (int i = 0; i < input % 1000; i++) { result += (long)Math.Sqrt(i * input); } return result; } }

线程同步

使用lock、Monitor和其他同步原语。

using System; using System.Threading; using System.Collections.Generic; using UnityEngine; public class ThreadSynchronization : MonoBehaviour { [Header("同步设置")] public int bufferSize = 10; private Queue<int> sharedBuffer; private readonly object bufferLock = new object(); private volatile bool isRunning = false; private int itemsProduced = 0; private int itemsConsumed = 0; void Start() { InitializeSynchronization(); Debug.Log("线程同步系统已初始化"); } void InitializeSynchronization() { sharedBuffer = new Queue<int>(bufferSize); } void Update() { if (Input.GetKeyDown(KeyCode.S)) { StartProducerConsumer(); } if (Input.GetKeyDown(KeyCode.T)) { StopProducerConsumer(); } if (Input.GetKeyDown(KeyCode.I)) { ShowSyncInfo(); } } public void StartProducerConsumer() { if (!isRunning) { isRunning = true; itemsProduced = 0; itemsConsumed = 0; Thread producer = new Thread(ProducerMethod); Thread consumer = new Thread(ConsumerMethod); producer.IsBackground = true; consumer.IsBackground = true; producer.Start(); consumer.Start(); Debug.Log("生产者和消费者已启动"); } } void ProducerMethod() { while (isRunning) { lock (bufferLock) { if (sharedBuffer.Count < bufferSize) { int item = UnityEngine.Random.Range(1, 1000); sharedBuffer.Enqueue(item); Interlocked.Increment(ref itemsProduced); Debug.Log($"生产项目 {item}。缓冲区: {sharedBuffer.Count}"); Monitor.PulseAll(bufferLock); } else { Monitor.Wait(bufferLock); } } Thread.Sleep(UnityEngine.Random.Range(100, 500)); } } void ConsumerMethod() { while (isRunning) { lock (bufferLock) { while (sharedBuffer.Count == 0 && isRunning) { Monitor.Wait(bufferLock); } if (sharedBuffer.Count > 0) { int item = sharedBuffer.Dequeue(); Interlocked.Increment(ref itemsConsumed); Debug.Log($"消费项目 {item}。缓冲区: {sharedBuffer.Count}"); Monitor.PulseAll(bufferLock); } } Thread.Sleep(UnityEngine.Random.Range(200, 800)); } } public void StopProducerConsumer() { if (isRunning) { isRunning = false; lock (bufferLock) { Monitor.PulseAll(bufferLock); } Debug.Log("正在停止生产者和消费者..."); } } public void ShowSyncInfo() { lock (bufferLock) { Debug.Log($"缓冲区: {sharedBuffer.Count}/{bufferSize}"); Debug.Log($"已生产: {itemsProduced}，已消费: {itemsConsumed}"); } } void OnDestroy() { StopProducerConsumer(); } }

Unity多线程

在Unity中安全使用线程和主线程交互。

using System; using System.Collections; using System.Collections.Concurrent; using System.Threading.Tasks; using UnityEngine; public class UnityThreadingManager : MonoBehaviour { [Header("Unity线程设置")] public Transform[] objectsToMove; private ConcurrentQueue<Action> mainThreadActions = new ConcurrentQueue<Action>(); private bool isProcessing = false; void Start() { Debug.Log("Unity线程管理器已初始化"); } void Update() { ProcessMainThreadActions(); if (Input.GetKeyDown(KeyCode.P)) { StartPathfindingJobs(); } if (Input.GetKeyDown(KeyCode.M)) { StartBackgroundCalculations(); } if (Input.GetKeyDown(KeyCode.L)) { StartLongRunningTask(); } } void ProcessMainThreadActions() { while (mainThreadActions.TryDequeue(out Action action)) { try { action?.Invoke(); } catch (Exception e) { Debug.LogError($"执行操作时出错: {e.Message}"); } } } public void StartPathfindingJobs() { if (isProcessing) return; isProcessing = true; Debug.Log("启动后台寻路任务..."); Task.Run(() => { Vector3 calculatedPath = CalculatePathInBackground(); mainThreadActions.Enqueue(() => { ApplyPathResult(calculatedPath); }); }); } Vector3 CalculatePathInBackground() { System.Threading.Thread.Sleep(2000); Vector3 result = new Vector3( UnityEngine.Random.Range(-10f, 10f), 0f, UnityEngine.Random.Range(-10f, 10f) ); Debug.Log($"后台任务在线程 {System.Threading.Thread.CurrentThread.ManagedThreadId} 上完成"); return result; } void ApplyPathResult(Vector3 newPath) { if (objectsToMove.Length > 0 && objectsToMove[0] != null) { StartCoroutine(MoveObjectToPosition(objectsToMove[0], newPath)); Debug.Log($"路径结果已应用: {newPath}"); } isProcessing = false; } IEnumerator MoveObjectToPosition(Transform obj, Vector3 targetPos) { Vector3 startPos = obj.position; float duration = 2f; float elapsed = 0f; while (elapsed < duration) { elapsed += Time.deltaTime; float t = elapsed / duration; obj.position = Vector3.Lerp(startPos, targetPos, t); yield return null; } obj.position = targetPos; } public void StartBackgroundCalculations() { Debug.Log("启动后台计算..."); Task.Run(async () => { for (int i = 0; i < 10; i++) { await Task.Delay(500); int progress = (i + 1) * 10; mainThreadActions.Enqueue(() => { Debug.Log($"进度: {progress}%"); }); } mainThreadActions.Enqueue(() => { Debug.Log("后台计算完成！"); }); }); } public async void StartLongRunningTask() { Debug.Log("启动长时间运行的任务..."); try { var result = await PerformLongRunningOperation(); Debug.Log($"长时间运行的任务完成: {result}"); } catch (Exception e) { Debug.LogError($"长时间运行的任务出错: {e.Message}"); } } private async Task<string> PerformLongRunningOperation() { return await Task.Run(() => { System.Threading.Thread.Sleep(3000); return $"操作在线程 {System.Threading.Thread.CurrentThread.ManagedThreadId} 上完成"; }); } void OnDestroy() { isProcessing = false; } }

高级线程技术

线程池、CancellationToken和性能优化。

using System; using System.Threading; using System.Threading.Tasks; using UnityEngine; public class AdvancedThreadingController : MonoBehaviour { [Header("高级设置")] public int maxConcurrentTasks = 4; private SemaphoreSlim semaphore; private CancellationTokenSource cancellationTokenSource; void Start() { InitializeAdvancedThreading(); Debug.Log("高级线程控制器已初始化"); } void InitializeAdvancedThreading() { semaphore = new SemaphoreSlim(maxConcurrentTasks, maxConcurrentTasks); cancellationTokenSource = new CancellationTokenSource(); } void Update() { if (Input.GetKeyDown(KeyCode.T)) { StartTaskPool(); } if (Input.GetKeyDown(KeyCode.C)) { CancelAllTasks(); } if (Input.GetKeyDown(KeyCode.R)) { StartReaderWriterExample(); } } public async void StartTaskPool() { Debug.Log("启动任务池..."); var tasks = new Task[10]; for (int i = 0; i < tasks.Length; i++) { int taskId = i; tasks[i] = ProcessTaskWithSemaphore(taskId, cancellationTokenSource.Token); } try { await Task.WhenAll(tasks); Debug.Log("任务池中的所有任务已完成"); } catch (OperationCanceledException) { Debug.Log("任务已被取消"); } } private async Task ProcessTaskWithSemaphore(int taskId, CancellationToken cancellationToken) { await semaphore.WaitAsync(cancellationToken); try { Debug.Log($"任务 {taskId} 已开始"); await Task.Delay(UnityEngine.Random.Range(1000, 3000), cancellationToken); Debug.Log($"任务 {taskId} 已完成"); } finally { semaphore.Release(); } } public void CancelAllTasks() { cancellationTokenSource.Cancel(); Debug.Log("正在取消所有任务..."); cancellationTokenSource = new CancellationTokenSource(); } private readonly ReaderWriterLockSlim readerWriterLock = new ReaderWriterLockSlim(); private int sharedResource = 0; public void StartReaderWriterExample() { Debug.Log("启动读写锁示例..."); // 启动读取器 for (int i = 0; i < 3; i++) { int readerId = i; Task.Run(() => ReaderTask(readerId)); } // 启动写入器 Task.Run(() => WriterTask()); } void ReaderTask(int readerId) { for (int i = 0; i < 5; i++) { readerWriterLock.EnterReadLock(); try { Debug.Log($"读取器 {readerId} 读取值: {sharedResource}"); Thread.Sleep(500); } finally { readerWriterLock.ExitReadLock(); } Thread.Sleep(200); } } void WriterTask() { for (int i = 0; i < 3; i++) { readerWriterLock.EnterWriteLock(); try { sharedResource++; Debug.Log($"写入器更新值为: {sharedResource}"); Thread.Sleep(1000); } finally { readerWriterLock.ExitWriteLock(); } Thread.Sleep(500); } } void OnDestroy() { cancellationTokenSource?.Cancel(); semaphore?.Dispose(); readerWriterLock?.Dispose(); } }
