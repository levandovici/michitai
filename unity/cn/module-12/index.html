模块12：区块链和Web3游戏

创建NFT、加密货币和智能合约游戏的完整示例

加密钱包集成

将MetaMask和其他Web3钱包连接到Unity游戏。

using UnityEngine; using System.Collections; public class WalletManager : MonoBehaviour { private string playerWalletAddress; private bool isWalletConnected = false; void Start() { Debug.Log("Web3钱包管理器已初始化"); } public void ConnectWallet() { Debug.Log("正在连接钱包..."); StartCoroutine(SimulateWalletConnection()); } IEnumerator SimulateWalletConnection() { yield return new WaitForSeconds(2f); playerWalletAddress = "0x742d35Cc6634C0532925a3b8D4C2C8b5C2C8b5C2"; isWalletConnected = true; Debug.Log($"钱包已连接: {playerWalletAddress}"); GetWalletBalance(); } IEnumerator FetchBalance() { yield return new WaitForSeconds(1f); float balance = Random.Range(0.1f, 5.0f); Debug.Log($"钱包余额: {balance:F4} ETH"); } public bool IsWalletConnected() { return isWalletConnected; } }

游戏内NFT系统

在游戏中创建、显示和使用NFT物品。

using UnityEngine; using System.Collections.Generic; [System.Serializable] public class NFTData { public int tokenId; public string name; public string rarity; public int attack = 10; public int defense = 10; } public class NFTManager : MonoBehaviour { private List<NFTData> playerNFTs = new List<NFTData>(); void Start() { CreateSampleNFTs(); Debug.Log("NFT管理器已初始化"); } void CreateSampleNFTs() { playerNFTs.Add(new NFTData { tokenId = 1, name = "火焰之剑", rarity = "传说", attack = 25, defense = 5 }); Debug.Log($"创建了{playerNFTs.Count}个NFT物品"); } public void MintNewNFT(string name, string rarity) { var newNFT = new NFTData { tokenId = playerNFTs.Count + 1, name = name, rarity = rarity, attack = Random.Range(10, 30), defense = Random.Range(10, 30) }; playerNFTs.Add(newNFT); Debug.Log($"新NFT已创建: {name}"); } }

智能合约交互

调用智能合约函数实现游戏逻辑。

using UnityEngine; using System.Collections; public class SmartContractManager : MonoBehaviour { public string gameContractAddress = "0x..."; void Start() { Debug.Log("智能合约管理器已初始化"); } public void ClaimReward(int amount) { StartCoroutine(ExecuteClaimReward(amount)); } IEnumerator ExecuteClaimReward(int amount) { Debug.Log($"智能合约调用: claimReward({amount})"); yield return new WaitForSeconds(3f); string txHash = "0x" + System.Guid.NewGuid().ToString("N").Substring(0, 16); Debug.Log($"奖励已领取！交易: {txHash}"); } public void PurchaseItem(int itemId, int price) { StartCoroutine(ExecutePurchaseItem(itemId, price)); } IEnumerator ExecutePurchaseItem(int itemId, int price) { Debug.Log($"购买物品{itemId}，价格{price}代币"); yield return new WaitForSeconds(2f); string txHash = "0x" + System.Guid.NewGuid().ToString("N").Substring(0, 16); Debug.Log($"物品已购买！交易: {txHash}"); } }

代币经济

游戏代币系统、质押和经济机制。

using UnityEngine; public class TokenManager : MonoBehaviour { public string tokenSymbol = "GAME"; private int currentBalance = 1000; private int stakedAmount = 0; void Start() { Debug.Log($"代币管理器已初始化。余额: {currentBalance} {tokenSymbol}"); } public void AddTokens(int amount) { currentBalance += amount; Debug.Log($"添加{amount} {tokenSymbol}。新余额: {currentBalance}"); } public bool SpendTokens(int amount) { if (currentBalance >= amount) { currentBalance -= amount; Debug.Log($"消费{amount} {tokenSymbol}。剩余: {currentBalance}"); return true; } Debug.LogWarning("代币不足"); return false; } public void StakeTokens(int amount) { if (currentBalance >= amount) { currentBalance -= amount; stakedAmount += amount; Debug.Log($"质押{amount} {tokenSymbol}"); } } public int GetBalance() { return currentBalance; } }

NFT市场

用于买卖NFT物品的交易平台。

using UnityEngine; using System.Collections.Generic; [System.Serializable] public class MarketplaceItem { public NFTData nft; public int price; public string seller; } public class MarketplaceManager : MonoBehaviour { private List<MarketplaceItem> marketplaceItems = new List<MarketplaceItem>(); void Start() { CreateSampleMarketplaceItems(); Debug.Log("NFT市场已初始化"); } void CreateSampleMarketplaceItems() { marketplaceItems.Add(new MarketplaceItem { nft = new NFTData { tokenId = 101, name = "冰霜法杖", rarity = "史诗" }, price = 500, seller = "0x123...456" }); Debug.Log($"在市场中创建了{marketplaceItems.Count}个物品"); } public void BuyItem(MarketplaceItem item) { var tokenManager = FindObjectOfType<TokenManager>(); if (tokenManager != null && tokenManager.SpendTokens(item.price)) { Debug.Log($"NFT已购买: {item.nft.name}，价格{item.price}代币"); } else { Debug.LogWarning("代币不足，无法购买"); } } public void SellItem(NFTData nft, int price) { var marketItem = new MarketplaceItem { nft = nft, price = price, seller = "Player" }; marketplaceItems.Add(marketItem); Debug.Log($"NFT {nft.name}已上架，价格{price}代币"); } }
