内存管理、跨平台开发和变现的完整示例

使用Addressables进行内存管理

高效加载和卸载资源以优化内存使用。

using UnityEngine; using UnityEngine.AddressableAssets; using UnityEngine.ResourceManagement.AsyncOperations; using System.Collections.Generic; public class AddressableManager : MonoBehaviour { public AssetReference playerPrefab; public AssetReference audioClip; private Dictionary<string, AsyncOperationHandle> loadedAssets = new Dictionary<string, AsyncOperationHandle>(); void Start() { LoadAssets(); Debug.Log("Addressable Manager 已初始化"); } async void LoadAssets() { // 加载玩家预制体 var playerHandle = playerPrefab.LoadAssetAsync<GameObject>(); await playerHandle.Task; if (playerHandle.Status == AsyncOperationStatus.Succeeded) { loadedAssets["player"] = playerHandle; var player = Instantiate(playerHandle.Result); Debug.Log("从Addressable资源创建玩家"); } // 加载音频 var audioHandle = audioClip.LoadAssetAsync<AudioClip>(); await audioHandle.Task; if (audioHandle.Status == AsyncOperationStatus.Succeeded) { loadedAssets["audio"] = audioHandle; PlayAudio(audioHandle.Result); } } void PlayAudio(AudioClip clip) { var audioSource = GetComponent<AudioSource>(); if (audioSource == null) audioSource = gameObject.AddComponent<AudioSource>(); audioSource.clip = clip; audioSource.Play(); Debug.Log("音频播放中"); } public void UnloadAsset(string key) { if (loadedAssets.ContainsKey(key)) { Addressables.Release(loadedAssets[key]); loadedAssets.Remove(key); Debug.Log($"资源 {key} 已卸载"); } } void OnDestroy() { foreach (var asset in loadedAssets.Values) { Addressables.Release(asset); } Debug.Log("所有Addressable资源已释放"); } }

跨平台开发

为不同平台适配游戏：Android、WebGL、PC。

using UnityEngine; public class PlatformManager : MonoBehaviour { public bool enableTouchControls = false; public float mobileUIScale = 1.2f; public float desktopUIScale = 1.0f; void Start() { DetectAndConfigurePlatform(); } void DetectAndConfigurePlatform() { switch (Application.platform) { case RuntimePlatform.Android: case RuntimePlatform.IPhonePlayer: SetupMobile(); break; case RuntimePlatform.WebGLPlayer: SetupWebGL(); break; default: SetupDesktop(); break; } Debug.Log($"平台: {Application.platform}"); } void SetupMobile() { enableTouchControls = true; SetUIScale(mobileUIScale); QualitySettings.SetQualityLevel(1); // 低质量 // 启用触摸元素 var touchControls = GameObject.FindGameObjectsWithTag("TouchControl"); foreach (var control in touchControls) { control.SetActive(true); } Debug.Log("移动端配置已应用"); } void SetupWebGL() { enableTouchControls = SystemInfo.deviceType == DeviceType.Handheld; SetUIScale(enableTouchControls ? mobileUIScale : desktopUIScale); QualitySettings.SetQualityLevel(2); // 中等质量 Debug.Log("WebGL配置已应用"); } void SetupDesktop() { enableTouchControls = false; SetUIScale(desktopUIScale); QualitySettings.SetQualityLevel(4); // 高质量 // 禁用触摸元素 var touchControls = GameObject.FindGameObjectsWithTag("TouchControl"); foreach (var control in touchControls) { control.SetActive(false); } Debug.Log("桌面端配置已应用"); } void SetUIScale(float scale) { var canvas = FindObjectOfType<Canvas>(); if (canvas != null) { var scaler = canvas.GetComponent<UnityEngine.UI.CanvasScaler>(); if (scaler != null) { scaler.scaleFactor = scale; } } } }

分析与追踪

收集玩家行为和游戏性能数据。

using UnityEngine; using System.Collections.Generic; public class AnalyticsManager : MonoBehaviour { public string gameVersion = "1.0.0"; private float sessionStartTime; private int levelCompletions = 0; void Start() { InitializeAnalytics(); TrackSessionStart(); } void InitializeAnalytics() { sessionStartTime = Time.time; Debug.Log("分析系统已初始化"); } void TrackSessionStart() { var data = new Dictionary<string, object>(); data["device"] = SystemInfo.deviceModel; data["os"] = SystemInfo.operatingSystem; data["version"] = gameVersion; data["platform"] = Application.platform.ToString(); TrackEvent("session_start", data); } public void TrackEvent(string eventName, Dictionary<string, object> parameters = null) { var eventData = new Dictionary<string, object>(); eventData["event"] = eventName; eventData["time"] = System.DateTime.Now.ToString(); eventData["session_time"] = Time.time - sessionStartTime; if (parameters != null) { foreach (var param in parameters) { eventData[param.Key] = param.Value; } } Debug.Log($"事件: {eventName}"); LogEvent(eventData); } public void TrackLevelComplete(int level, float time) { levelCompletions++; var data = new Dictionary<string, object>(); data["level"] = level; data["completion_time"] = time; data["total_completions"] = levelCompletions; TrackEvent("level_complete", data); } public void TrackPurchase(string item, float price) { var data = new Dictionary<string, object>(); data["item"] = item; data["price"] = price; TrackEvent("purchase", data); } void LogEvent(Dictionary<string, object> data) { string log = "分析数据:\n"; foreach (var item in data) { log += $" {item.Key}: {item.Value}\n"; } Debug.Log(log); } void OnDestroy() { var sessionTime = Time.time - sessionStartTime; var data = new Dictionary<string, object>(); data["duration"] = sessionTime; data["completions"] = levelCompletions; TrackEvent("session_end", data); } }

使用Unity Ads进行变现

集成广告和内购实现游戏变现。

using UnityEngine; using UnityEngine.UI; public class MonetizationManager : MonoBehaviour { [Header("设置")] public string gameId = "1234567"; public Button watchAdButton; public Text coinsText; private int playerCoins = 0; private bool adReady = false; void Start() { InitializeMonetization(); LoadPlayerData(); } void InitializeMonetization() { // 模拟Unity Ads初始化 Invoke("SetAdReady", 2f); if (watchAdButton != null) { watchAdButton.onClick.AddListener(ShowRewardedAd); } Debug.Log("变现系统已初始化"); } void SetAdReady() { adReady = true; UpdateUI(); Debug.Log("广告已准备就绪"); } void LoadPlayerData() { playerCoins = PlayerPrefs.GetInt("Coins", 0); UpdateUI(); } void SavePlayerData() { PlayerPrefs.SetInt("Coins", playerCoins); PlayerPrefs.Save(); } public void ShowRewardedAd() { if (!adReady) return; Debug.Log("显示广告中..."); adReady = false; UpdateUI(); // 模拟观看广告 Invoke("OnAdComplete", 3f); } void OnAdComplete() { int reward = 50; playerCoins += reward; SavePlayerData(); UpdateUI(); Debug.Log($"观看广告获得 {reward} 金币"); // 30秒后下一个广告 Invoke("SetAdReady", 30f); } public void PurchaseCoins(int amount, float price) { playerCoins += amount; SavePlayerData(); UpdateUI(); // 发送事件到分析系统 var analytics = FindObjectOfType<AnalyticsManager>(); if (analytics != null) { analytics.TrackPurchase($"coins_{amount}", price); } Debug.Log($"购买了 {amount} 金币，花费 ${price}"); } public bool SpendCoins(int amount) { if (playerCoins >= amount) { playerCoins -= amount; SavePlayerData(); UpdateUI(); Debug.Log($"消费了 {amount} 金币"); return true; } Debug.Log("金币不足"); return false; } void UpdateUI() { if (coinsText != null) { coinsText.text = $"金币: {playerCoins}"; } if (watchAdButton != null) { watchAdButton.interactable = adReady; } } }

发布准备

自动化构建流程并为游戏发布做准备。

using UnityEngine; #if UNITY_EDITOR using UnityEditor; using UnityEditor.Build.Reporting; #endif public class BuildManager : MonoBehaviour { [Header("构建设置")] public string productName = "MyGame"; public string version = "1.0.0"; #if UNITY_EDITOR [ContextMenu("构建Android")] public void BuildAndroid() { SetupBuildSettings(); var buildOptions = new BuildPlayerOptions(); buildOptions.scenes = GetScenePaths(); buildOptions.locationPathName = $"Builds/Android/{productName}.apk"; buildOptions.target = BuildTarget.Android; buildOptions.options = BuildOptions.None; Debug.Log("开始Android构建..."); var report = BuildPipeline.BuildPlayer(buildOptions); if (report.summary.result == BuildResult.Succeeded) { Debug.Log("Android构建成功完成！"); } else { Debug.LogError("Android构建失败"); } } [ContextMenu("构建Windows")] public void BuildWindows() { SetupBuildSettings(); var buildOptions = new BuildPlayerOptions(); buildOptions.scenes = GetScenePaths(); buildOptions.locationPathName = $"Builds/Windows/{productName}.exe"; buildOptions.target = BuildTarget.StandaloneWindows64; buildOptions.options = BuildOptions.None; Debug.Log("开始Windows构建..."); var report = BuildPipeline.BuildPlayer(buildOptions); if (report.summary.result == BuildResult.Succeeded) { Debug.Log("Windows构建成功完成！"); } } void SetupBuildSettings() { PlayerSettings.productName = productName; PlayerSettings.bundleVersion = version; // Android设置 PlayerSettings.Android.bundleVersionCode++; PlayerSettings.Android.minSdkVersion = AndroidSdkVersions.AndroidApiLevel21; Debug.Log($"构建设置: {productName} v{version}"); } string[] GetScenePaths() { var scenes = new string[EditorBuildSettings.scenes.Length]; for (int i = 0; i < scenes.Length; i++) { scenes[i] = EditorBuildSettings.scenes[i].path; } return scenes; } [ContextMenu("应用商店准备")] public void PrepareForStore() { // 设置应用商店配置 PlayerSettings.Android.useAPKExpansionFiles = false; PlayerSettings.Android.keystoreName = "user.keystore"; Debug.Log("游戏已准备好发布到应用商店"); } #endif }
