Exemplul 1: Variabile și tipuri de date

Tipurile de date principale și lucrul cu variabile în C#:

using UnityEngine;

public class VariablesExample : MonoBehaviour
{
    // Tipuri numerice
    public int playerHealth = 100;
    public float moveSpeed = 5.5f;
    public double preciseValue = 3.14159;

    // Text și logică
    public string playerName = "Jucător";
    public bool isAlive = true;
    public char grade = 'A';

    // Tipuri Unity
    public Vector3 position;
    public Color playerColor = Color.red;

    void Start()
    {
        ShowVariableTypes();
        PerformCalculations();
    }

    void ShowVariableTypes()
    {
        Debug.Log("Viață (int): " + playerHealth);
        Debug.Log("Viteză (float): " + moveSpeed);
        Debug.Log("Nume (string): " + playerName);
        Debug.Log("În viață (bool): " + isAlive);
    }

    void PerformCalculations()
    {
        int damage = 25;
        int newHealth = playerHealth - damage;
        Debug.Log($"Viață după damage: {newHealth}");

        float distance = moveSpeed * Time.deltaTime;
        Debug.Log($"Distanță pe frame: {distance}");

        bool isHealthLow = playerHealth < 30;
        Debug.Log($"Viață scăzută? {isHealthLow}");
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.H))
        {
            playerHealth -= 10;
            Debug.Log("Damage! Viață: " + playerHealth);
        }
    }
}

Exemplul 2: Funcții și metode

Crearea și utilizarea funcțiilor în C#:

using UnityEngine;

public class FunctionsExample : MonoBehaviour
{
    private int currentHealth = 100;
    private int experience = 0;

    void Start()
    {
        InitializePlayer();
        ShowPlayerInfo();
    }

    // Funcție fără parametri
    void InitializePlayer()
    {
        currentHealth = 100;
        experience = 0;
        Debug.Log("Jucător inițializat");
    }

    // Funcție cu parametri
    void TakeDamage(int damage)
    {
        currentHealth -= damage;
        currentHealth = Mathf.Max(0, currentHealth);
        Debug.Log($"Damage: {damage}. Viață: {currentHealth}");
    }

    // Funcție cu valoare returnată
    bool IsAlive()
    {
        return currentHealth > 0;
    }

    // Funcție cu mai mulți parametri
    float CalculateDamage(float baseDamage, float multiplier)
    {
        return baseDamage * multiplier;
    }

    // Funcție cu parametru implicit
    void Heal(int healAmount = 20)
    {
        currentHealth += healAmount;
        Debug.Log($"Vindecare: {healAmount}. Viață: {currentHealth}");
    }

    public void ShowPlayerInfo()
    {
        Debug.Log($"Viață: {currentHealth}");
        Debug.Log($"Experiență: {experience}");
        Debug.Log($"În viață: {IsAlive()}");
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.Alpha1))
        {
            float damage = CalculateDamage(20f, 1.5f);
            TakeDamage(Mathf.RoundToInt(damage));
        }

        if (Input.GetKeyDown(KeyCode.Alpha2))
        {
            Heal();
        }
    }
}

Exemplul 3: Clase și obiecte

Crearea și utilizarea claselor în C#:

using UnityEngine;

// Clasa de bază pentru personaje
public class Character
{
    protected string characterName;
    protected int health;
    protected int maxHealth;

    public string Name { get { return characterName; } }
    public int Health { get { return health; } }
    public bool IsAlive => health > 0;

    public Character(string name, int maxHp)
    {
        characterName = name;
        maxHealth = maxHp;
        health = maxHealth;
    }

    public virtual void TakeDamage(int damage)
    {
        health -= damage;
        health = Mathf.Max(0, health);
        Debug.Log($"{characterName} a primit {damage} damage");
    }
}

// Clasa jucător
public class Player : Character
{
    private int experience;

    public Player(string name) : base(name, 100)
    {
        experience = 0;
    }

    public override void TakeDamage(int damage)
    {
        int reducedDamage = Mathf.Max(1, damage - 5);
        base.TakeDamage(reducedDamage);
        Debug.Log("Armura a absorbit damage!");
    }

    public void AddExperience(int exp)
    {
        experience += exp;
        Debug.Log($"Experiență primită: {exp}");
    }
}

// Clasa inamic
public class Enemy : Character
{
    private int attackPower;

    public Enemy(string name, int hp, int attack) : base(name, hp)
    {
        attackPower = attack;
    }

    public void Attack(Character target)
    {
        Debug.Log($"{characterName} atacă!");
        target.TakeDamage(attackPower);
    }
}

public class ClassesExample : MonoBehaviour
{
    private Player player;
    private Enemy enemy;

    void Start()
    {
        player = new Player("Erou");
        enemy = new Enemy("Goblin", 50, 15);
        
        Debug.Log("Personaje create!");
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.Alpha1))
        {
            player.TakeDamage(10);
        }

        if (Input.GetKeyDown(KeyCode.Alpha2))
        {
            enemy.Attack(player);
        }
    }
}

Exemplul 4: Array-uri și liste

Lucrul cu array-uri și colecții în C#:

using UnityEngine;
using System.Collections.Generic;

public class ArraysExample : MonoBehaviour
{
    // Array-uri
    public string[] playerNames = {"Alexandru", "Maria", "Ion"};
    public int[] scores = {100, 250, 180};

    // Liste
    public List<string> inventory = new List<string>();

    void Start()
    {
        DemonstrateArrays();
        DemonstrateLists();
    }

    void DemonstrateArrays()
    {
        Debug.Log("=== Array-uri ===");
        
        // Crearea unui array
        int[] numbers = new int[5];
        for (int i = 0; i < numbers.Length; i++)
        {
            numbers[i] = i * 10;
        }

        // Afișarea array-ului
        foreach (int number in numbers)
        {
            Debug.Log("Număr: " + number);
        }

        // Lucrul cu nume și scoruri
        for (int i = 0; i < playerNames.Length; i++)
        {
            Debug.Log($"{playerNames[i]} - {scores[i]} puncte");
        }
    }

    void DemonstrateLists()
    {
        Debug.Log("=== Liste ===");
        
        // Adăugarea elementelor
        inventory.Add("Spadă");
        inventory.Add("Scut");
        inventory.Add("Poțiune");

        // Afișarea listei
        for (int i = 0; i < inventory.Count; i++)
        {
            Debug.Log($"{i + 1}. {inventory[i]}");
        }

        // Căutarea și eliminarea
        bool hasSword = inventory.Contains("Spadă");
        Debug.Log("Are spadă: " + hasSword);

        inventory.Remove("Poțiune");
        Debug.Log("Poțiune folosită");
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.Alpha1))
        {
            string[] items = {"Inel", "Amuletă", "Pergament"};
            string randomItem = items[Random.Range(0, items.Length)];
            inventory.Add(randomItem);
            Debug.Log("Găsit: " + randomItem);
        }
    }
}

Exemplul 5: Operatori condiționale

Utilizarea condițiilor și buclelor în C#:

using UnityEngine;

public class ConditionsExample : MonoBehaviour
{
    public int playerLevel = 1;
    public int playerHealth = 100;
    public string playerClass = "Războinic";

    void Start()
    {
        DemonstrateConditions();
        DemonstrateLoops();
    }

    void DemonstrateConditions()
    {
        Debug.Log("=== Condiții ===");

        // Condiție simplă if
        if (playerHealth > 50)
        {
            Debug.Log("Viața este OK");
        }

        // if-else
        if (playerLevel >= 10)
        {
            Debug.Log("Jucător experimentat");
        }
        else
        {
            Debug.Log("Începător");
        }

        // if-else if-else
        if (playerHealth > 80)
        {
            Debug.Log("Viață excelentă");
        }
        else if (playerHealth > 50)
        {
            Debug.Log("Viață bună");
        }
        else if (playerHealth > 20)
        {
            Debug.Log("Viață scăzută");
        }
        else
        {
            Debug.Log("Stare critică!");
        }

        // switch
        switch (playerClass)
        {
            case "Războinic":
                Debug.Log("Puternic în luptă apropiată");
                break;
            case "Mag":
                Debug.Log("Stăpânește magia");
                break;
            case "Arcaș":
                Debug.Log("Maestru în lupta la distanță");
                break;
            default:
                Debug.Log("Clasă necunoscută");
                break;
        }
    }

    void DemonstrateLoops()
    {
        Debug.Log("=== Bucle ===");

        // Bucla for
        Debug.Log("Bucla for:");
        for (int i = 1; i <= 5; i++)
        {
            Debug.Log("Iterația: " + i);
        }

        // Bucla while
        Debug.Log("Bucla while:");
        int count = 0;
        while (count < 3)
        {
            Debug.Log("Count: " + count);
            count++;
        }

        // Bucla foreach cu array
        string[] weapons = {"Spadă", "Arc", "Toiag"};
        Debug.Log("Arme:");
        foreach (string weapon in weapons)
        {
            Debug.Log("- " + weapon);
        }
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.Alpha1))
        {
            playerHealth -= 20;
            CheckHealthStatus();
        }

        if (Input.GetKeyDown(KeyCode.Alpha2))
        {
            playerLevel++;
            CheckLevelRewards();
        }
    }

    void CheckHealthStatus()
    {
        if (playerHealth <= 0)
        {
            Debug.Log("Jucătorul a murit!");
        }
        else if (playerHealth < 30)
        {
            Debug.Log("Necesită vindecare!");
        }
    }

    void CheckLevelRewards()
    {
        if (playerLevel % 5 == 0)
        {
            Debug.Log("Recompensă pentru nivel!");
        }
    }
}
