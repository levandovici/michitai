Modulul 12: Blockchain și Jocuri Web3

Exemple complete pentru crearea jocurilor cu NFT, criptomonede și contracte smart

Integrarea Portofelului Cripto

Conectarea MetaMask și a altor portofele Web3 la jocurile Unity.

using UnityEngine; using System.Collections; public class WalletManager : MonoBehaviour { private string playerWalletAddress; private bool isWalletConnected = false; void Start() { Debug.Log("Web3 Wallet Manager inițializat"); } public void ConnectWallet() { Debug.Log("Conectare la portofel..."); StartCoroutine(SimulateWalletConnection()); } IEnumerator SimulateWalletConnection() { yield return new WaitForSeconds(2f); playerWalletAddress = "0x742d35Cc6634C0532925a3b8D4C2C8b5C2C8b5C2"; isWalletConnected = true; Debug.Log($"Portofel conectat: {playerWalletAddress}"); GetWalletBalance(); } IEnumerator FetchBalance() { yield return new WaitForSeconds(1f); float balance = Random.Range(0.1f, 5.0f); Debug.Log($"Sold portofel: {balance:F4} ETH"); } public bool IsWalletConnected() { return isWalletConnected; } }

Sistemul NFT în Joc

Crearea, afișarea și utilizarea obiectelor NFT în jocuri.

using UnityEngine; using System.Collections.Generic; [System.Serializable] public class NFTData { public int tokenId; public string name; public string rarity; public int attack = 10; public int defense = 10; } public class NFTManager : MonoBehaviour { private List<NFTData> playerNFTs = new List<NFTData>(); void Start() { CreateSampleNFTs(); Debug.Log("NFT Manager inițializat"); } void CreateSampleNFTs() { playerNFTs.Add(new NFTData { tokenId = 1, name = "Sabia de Foc", rarity = "Legendar", attack = 25, defense = 5 }); Debug.Log($"Creat {playerNFTs.Count} obiecte NFT"); } public void MintNewNFT(string name, string rarity) { var newNFT = new NFTData { tokenId = playerNFTs.Count + 1, name = name, rarity = rarity, attack = Random.Range(10, 30), defense = Random.Range(10, 30) }; playerNFTs.Add(newNFT); Debug.Log($"NFT nou creat: {name}"); } }

Interacțiunea cu Contractele Smart

Apelarea funcțiilor contractelor smart pentru logica jocului.

using UnityEngine; using System.Collections; public class SmartContractManager : MonoBehaviour { public string gameContractAddress = "0x..."; void Start() { Debug.Log("Smart Contract Manager inițializat"); } public void ClaimReward(int amount) { StartCoroutine(ExecuteClaimReward(amount)); } IEnumerator ExecuteClaimReward(int amount) { Debug.Log($"Apel contract smart: claimReward({amount})"); yield return new WaitForSeconds(3f); string txHash = "0x" + System.Guid.NewGuid().ToString("N").Substring(0, 16); Debug.Log($"Recompensă primită! Tranzacție: {txHash}"); } public void PurchaseItem(int itemId, int price) { StartCoroutine(ExecutePurchaseItem(itemId, price)); } IEnumerator ExecutePurchaseItem(int itemId, int price) { Debug.Log($"Cumpărare obiect {itemId} pentru {price} tokeni"); yield return new WaitForSeconds(2f); string txHash = "0x" + System.Guid.NewGuid().ToString("N").Substring(0, 16); Debug.Log($"Obiect cumpărat! Tranzacție: {txHash}"); } }

Economia de Tokeni

Sistemul de tokeni de joc, staking și mecanici economice.

using UnityEngine; public class TokenManager : MonoBehaviour { public string tokenSymbol = "GAME"; private int currentBalance = 1000; private int stakedAmount = 0; void Start() { Debug.Log($"Token Manager inițializat. Sold: {currentBalance} {tokenSymbol}"); } public void AddTokens(int amount) { currentBalance += amount; Debug.Log($"Adăugat {amount} {tokenSymbol}. Sold nou: {currentBalance}"); } public bool SpendTokens(int amount) { if (currentBalance >= amount) { currentBalance -= amount; Debug.Log($"Cheltuit {amount} {tokenSymbol}. Rămas: {currentBalance}"); return true; } Debug.LogWarning("Tokeni insuficienți"); return false; } public void StakeTokens(int amount) { if (currentBalance >= amount) { currentBalance -= amount; stakedAmount += amount; Debug.Log($"Staked {amount} {tokenSymbol}"); } } public int GetBalance() { return currentBalance; } }

Marketplace NFT

Platformă de tranzacționare pentru cumpărarea și vânzarea obiectelor NFT.

using UnityEngine; using System.Collections.Generic; [System.Serializable] public class MarketplaceItem { public NFTData nft; public int price; public string seller; } public class MarketplaceManager : MonoBehaviour { private List<MarketplaceItem> marketplaceItems = new List<MarketplaceItem>(); void Start() { CreateSampleMarketplaceItems(); Debug.Log("NFT Marketplace inițializat"); } void CreateSampleMarketplaceItems() { marketplaceItems.Add(new MarketplaceItem { nft = new NFTData { tokenId = 101, name = "Toiag de Gheață", rarity = "Epic" }, price = 500, seller = "0x123...456" }); Debug.Log($"Creat {marketplaceItems.Count} obiecte în marketplace"); } public void BuyItem(MarketplaceItem item) { var tokenManager = FindObjectOfType<TokenManager>(); if (tokenManager != null && tokenManager.SpendTokens(item.price)) { Debug.Log($"NFT cumpărat: {item.nft.name} pentru {item.price} tokeni"); } else { Debug.LogWarning("Tokeni insuficienți pentru cumpărare"); } } public void SellItem(NFTData nft, int price) { var marketItem = new MarketplaceItem { nft = nft, price = price, seller = "Player" }; marketplaceItems.Add(marketItem); Debug.Log($"NFT {nft.name} pus la vânzare pentru {price} tokeni"); } }
