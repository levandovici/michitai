Modulul 7: Programare Multithreading

Exemple cuprinzătoare de multithreading și programare paralelă în Unity

Bazele lucrului cu thread-uri

Introducere în thread-uri și operații de bază cu clasa Thread.

using System; using System.Threading; using UnityEngine; public class ThreadingBasics : MonoBehaviour { [Header("Setări Thread-uri")] public int numberOfThreads = 4; public bool useBackgroundThreads = true; private Thread[] workerThreads; private bool isRunning = false; void Start() { InitializeThreads(); Debug.Log("Sistemul de thread-uri a fost inițializat"); } void InitializeThreads() { workerThreads = new Thread[numberOfThreads]; for (int i = 0; i < numberOfThreads; i++) { int threadId = i; workerThreads[i] = new Thread(() => WorkerThreadMethod(threadId)); workerThreads[i].IsBackground = useBackgroundThreads; workerThreads[i].Name = $"Thread de lucru {threadId}"; } } void WorkerThreadMethod(int threadId) { Debug.Log($"Thread-ul {threadId} pornit pe nucleul {Thread.CurrentThread.ManagedThreadId}"); while (isRunning) { PerformWork(threadId); Thread.Sleep(1000); } Debug.Log($"Thread-ul {threadId} s-a terminat"); } void PerformWork(int threadId) { int result = 0; for (int i = 0; i < 1000000; i++) { result += i % (threadId + 1); } Debug.Log($"Thread-ul {threadId} a terminat munca. Rezultat: {result}"); } void Update() { if (Input.GetKeyDown(KeyCode.S)) { StartThreads(); } if (Input.GetKeyDown(KeyCode.T)) { StopThreads(); } } public void StartThreads() { if (!isRunning) { isRunning = true; for (int i = 0; i < workerThreads.Length; i++) { if (workerThreads[i].ThreadState == ThreadState.Unstarted) { workerThreads[i].Start(); } } Debug.Log("Toate thread-urile au fost pornite"); } } public void StopThreads() { if (isRunning) { isRunning = false; Debug.Log("Oprirea thread-urilor..."); } } void OnDestroy() { StopThreads(); } }

Task Parallel Library

Utilizarea API-ului modern pentru programarea paralelă.

using System; using System.Collections.Generic; using System.Threading.Tasks; using UnityEngine; public class TaskParallelController : MonoBehaviour { [Header("Setări Task-uri")] public int dataSize = 1000000; private List<int> largeDataSet; private CancellationTokenSource cancellationTokenSource; void Start() { InitializeData(); Debug.Log("Task Parallel Library Controller inițializat"); } void InitializeData() { largeDataSet = new List<int>(dataSize); for (int i = 0; i < dataSize; i++) { largeDataSet.Add(UnityEngine.Random.Range(1, 1000)); } Debug.Log($"Set de date creat cu {dataSize} elemente"); } void Update() { if (Input.GetKeyDown(KeyCode.P)) { ProcessDataParallel(); } if (Input.GetKeyDown(KeyCode.S)) { ProcessDataSequential(); } if (Input.GetKeyDown(KeyCode.A)) { ProcessDataAsync(); } } public void ProcessDataParallel() { Debug.Log("Începerea procesării paralele..."); var stopwatch = System.Diagnostics.Stopwatch.StartNew(); long sum = 0; object lockObject = new object(); Parallel.ForEach(largeDataSet, (number) => { long result = ComputeExpensiveOperation(number); lock (lockObject) { sum += result; } }); stopwatch.Stop(); Debug.Log($"Procesarea paralelă terminată în {stopwatch.ElapsedMilliseconds} ms. Suma: {sum}"); } public void ProcessDataSequential() { Debug.Log("Începerea procesării secvențiale..."); var stopwatch = System.Diagnostics.Stopwatch.StartNew(); long sum = 0; foreach (int number in largeDataSet) { sum += ComputeExpensiveOperation(number); } stopwatch.Stop(); Debug.Log($"Procesarea secvențială terminată în {stopwatch.ElapsedMilliseconds} ms. Suma: {sum}"); } public async void ProcessDataAsync() { Debug.Log("Începerea procesării asincrone..."); cancellationTokenSource = new CancellationTokenSource(); try { var result = await ProcessDataAsyncInternal(cancellationTokenSource.Token); Debug.Log($"Procesarea asincronă terminată. Rezultat: {result}"); } catch (OperationCanceledException) { Debug.Log("Operația asincronă a fost anulată"); } } private async Task<long> ProcessDataAsyncInternal(CancellationToken cancellationToken) { return await Task.Run(() => { long sum = 0; for (int i = 0; i < largeDataSet.Count; i++) { cancellationToken.ThrowIfCancellationRequested(); sum += ComputeExpensiveOperation(largeDataSet[i]); } return sum; }, cancellationToken); } private long ComputeExpensiveOperation(int input) { long result = 0; for (int i = 0; i < input % 1000; i++) { result += (long)Math.Sqrt(i * input); } return result; } }

Sincronizarea thread-urilor

Utilizarea lock, Monitor și altor primitive de sincronizare.

using System; using System.Threading; using System.Collections.Generic; using UnityEngine; public class ThreadSynchronization : MonoBehaviour { [Header("Setări Sincronizare")] public int bufferSize = 10; private Queue<int> sharedBuffer; private readonly object bufferLock = new object(); private volatile bool isRunning = false; private int itemsProduced = 0; private int itemsConsumed = 0; void Start() { InitializeSynchronization(); Debug.Log("Sistemul de sincronizare thread-uri inițializat"); } void InitializeSynchronization() { sharedBuffer = new Queue<int>(bufferSize); } void Update() { if (Input.GetKeyDown(KeyCode.S)) { StartProducerConsumer(); } if (Input.GetKeyDown(KeyCode.T)) { StopProducerConsumer(); } if (Input.GetKeyDown(KeyCode.I)) { ShowSyncInfo(); } } public void StartProducerConsumer() { if (!isRunning) { isRunning = true; itemsProduced = 0; itemsConsumed = 0; Thread producer = new Thread(ProducerMethod); Thread consumer = new Thread(ConsumerMethod); producer.IsBackground = true; consumer.IsBackground = true; producer.Start(); consumer.Start(); Debug.Log("Producătorul și consumatorul au fost porniți"); } } void ProducerMethod() { while (isRunning) { lock (bufferLock) { if (sharedBuffer.Count < bufferSize) { int item = UnityEngine.Random.Range(1, 1000); sharedBuffer.Enqueue(item); Interlocked.Increment(ref itemsProduced); Debug.Log($"Element produs {item}. Buffer: {sharedBuffer.Count}"); Monitor.PulseAll(bufferLock); } else { Monitor.Wait(bufferLock); } } Thread.Sleep(UnityEngine.Random.Range(100, 500)); } } void ConsumerMethod() { while (isRunning) { lock (bufferLock) { while (sharedBuffer.Count == 0 && isRunning) { Monitor.Wait(bufferLock); } if (sharedBuffer.Count > 0) { int item = sharedBuffer.Dequeue(); Interlocked.Increment(ref itemsConsumed); Debug.Log($"Element procesat {item}. Buffer: {sharedBuffer.Count}"); Monitor.PulseAll(bufferLock); } } Thread.Sleep(UnityEngine.Random.Range(200, 800)); } } public void StopProducerConsumer() { if (isRunning) { isRunning = false; lock (bufferLock) { Monitor.PulseAll(bufferLock); } Debug.Log("Oprirea producătorului și consumatorului..."); } } public void ShowSyncInfo() { lock (bufferLock) { Debug.Log($"Buffer: {sharedBuffer.Count}/{bufferSize}"); Debug.Log($"Produs: {itemsProduced}, Consumat: {itemsConsumed}"); } } void OnDestroy() { StopProducerConsumer(); } }

Multithreading în Unity

Lucrul sigur cu thread-uri în Unity și interacțiunea cu thread-ul principal.

using System; using System.Collections; using System.Collections.Concurrent; using System.Threading.Tasks; using UnityEngine; public class UnityThreadingManager : MonoBehaviour { [Header("Setări Unity Threading")] public Transform[] objectsToMove; private ConcurrentQueue<Action> mainThreadActions = new ConcurrentQueue<Action>(); private bool isProcessing = false; void Start() { Debug.Log("Unity Threading Manager inițializat"); } void Update() { ProcessMainThreadActions(); if (Input.GetKeyDown(KeyCode.P)) { StartPathfindingJobs(); } if (Input.GetKeyDown(KeyCode.M)) { StartBackgroundCalculations(); } if (Input.GetKeyDown(KeyCode.L)) { StartLongRunningTask(); } } void ProcessMainThreadActions() { while (mainThreadActions.TryDequeue(out Action action)) { try { action?.Invoke(); } catch (Exception e) { Debug.LogError($"Eroare la executarea acțiunii: {e.Message}"); } } } public void StartPathfindingJobs() { if (isProcessing) return; isProcessing = true; Debug.Log("Pornirea task-urilor de căutare cale în fundal..."); Task.Run(() => { Vector3 calculatedPath = CalculatePathInBackground(); mainThreadActions.Enqueue(() => { ApplyPathResult(calculatedPath); }); }); } Vector3 CalculatePathInBackground() { System.Threading.Thread.Sleep(2000); Vector3 result = new Vector3( UnityEngine.Random.Range(-10f, 10f), 0f, UnityEngine.Random.Range(-10f, 10f) ); Debug.Log($"Task de fundal terminat pe thread-ul {System.Threading.Thread.CurrentThread.ManagedThreadId}"); return result; } void ApplyPathResult(Vector3 newPath) { if (objectsToMove.Length > 0 && objectsToMove[0] != null) { StartCoroutine(MoveObjectToPosition(objectsToMove[0], newPath)); Debug.Log($"Rezultatul căii aplicat: {newPath}"); } isProcessing = false; } IEnumerator MoveObjectToPosition(Transform obj, Vector3 targetPos) { Vector3 startPos = obj.position; float duration = 2f; float elapsed = 0f; while (elapsed < duration) { elapsed += Time.deltaTime; float t = elapsed / duration; obj.position = Vector3.Lerp(startPos, targetPos, t); yield return null; } obj.position = targetPos; } public void StartBackgroundCalculations() { Debug.Log("Pornirea calculelor de fundal..."); Task.Run(async () => { for (int i = 0; i < 10; i++) { await Task.Delay(500); int progress = (i + 1) * 10; mainThreadActions.Enqueue(() => { Debug.Log($"Progres: {progress}%"); }); } mainThreadActions.Enqueue(() => { Debug.Log("Calculele de fundal terminate!"); }); }); } public async void StartLongRunningTask() { Debug.Log("Pornirea task-ului de lungă durată..."); try { var result = await PerformLongRunningOperation(); Debug.Log($"Task-ul de lungă durată terminat: {result}"); } catch (Exception e) { Debug.LogError($"Eroare în task-ul de lungă durată: {e.Message}"); } } private async Task<string> PerformLongRunningOperation() { return await Task.Run(() => { System.Threading.Thread.Sleep(3000); return $"Operația terminată pe thread-ul {System.Threading.Thread.CurrentThread.ManagedThreadId}"; }); } void OnDestroy() { isProcessing = false; } }

Tehnici avansate de multithreading

Pool-uri de thread-uri, CancellationToken și optimizarea performanței.

using System; using System.Threading; using System.Threading.Tasks; using UnityEngine; public class AdvancedThreadingController : MonoBehaviour { [Header("Setări Avansate")] public int maxConcurrentTasks = 4; private SemaphoreSlim semaphore; private CancellationTokenSource cancellationTokenSource; void Start() { InitializeAdvancedThreading(); Debug.Log("Advanced Threading Controller inițializat"); } void InitializeAdvancedThreading() { semaphore = new SemaphoreSlim(maxConcurrentTasks, maxConcurrentTasks); cancellationTokenSource = new CancellationTokenSource(); } void Update() { if (Input.GetKeyDown(KeyCode.T)) { StartTaskPool(); } if (Input.GetKeyDown(KeyCode.C)) { CancelAllTasks(); } if (Input.GetKeyDown(KeyCode.R)) { StartReaderWriterExample(); } } public async void StartTaskPool() { Debug.Log("Pornirea pool-ului de task-uri..."); var tasks = new Task[10]; for (int i = 0; i < tasks.Length; i++) { int taskId = i; tasks[i] = ProcessTaskWithSemaphore(taskId, cancellationTokenSource.Token); } try { await Task.WhenAll(tasks); Debug.Log("Toate task-urile din pool terminate"); } catch (OperationCanceledException) { Debug.Log("Task-urile au fost anulate"); } } private async Task ProcessTaskWithSemaphore(int taskId, CancellationToken cancellationToken) { await semaphore.WaitAsync(cancellationToken); try { Debug.Log($"Task-ul {taskId} început"); await Task.Delay(UnityEngine.Random.Range(1000, 3000), cancellationToken); Debug.Log($"Task-ul {taskId} terminat"); } finally { semaphore.Release(); } } public void CancelAllTasks() { cancellationTokenSource.Cancel(); Debug.Log("Anularea tuturor task-urilor..."); cancellationTokenSource = new CancellationTokenSource(); } private readonly ReaderWriterLockSlim readerWriterLock = new ReaderWriterLockSlim(); private int sharedResource = 0; public void StartReaderWriterExample() { Debug.Log("Pornirea exemplului ReaderWriterLock..."); // Pornirea cititorilor for (int i = 0; i < 3; i++) { int readerId = i; Task.Run(() => ReaderTask(readerId)); } // Pornirea scriitorului Task.Run(() => WriterTask()); } void ReaderTask(int readerId) { for (int i = 0; i < 5; i++) { readerWriterLock.EnterReadLock(); try { Debug.Log($"Cititorul {readerId} a citit valoarea: {sharedResource}"); Thread.Sleep(500); } finally { readerWriterLock.ExitReadLock(); } Thread.Sleep(200); } } void WriterTask() { for (int i = 0; i < 3; i++) { readerWriterLock.EnterWriteLock(); try { sharedResource++; Debug.Log($"Scriitorul a actualizat valoarea la: {sharedResource}"); Thread.Sleep(1000); } finally { readerWriterLock.ExitWriteLock(); } Thread.Sleep(500); } } void OnDestroy() { cancellationTokenSource?.Cancel(); semaphore?.Dispose(); readerWriterLock?.Dispose(); } }
