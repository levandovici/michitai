Exemplu: Conectarea portofelului Web3

Acest exemplu arată cum să conectezi un portofel MetaMask la un joc Unity:

using System;
using System.Threading.Tasks;
using UnityEngine;
using Nethereum.Web3;

/// 
/// Conectarea la portofelul Web3 (MetaMask) din Unity
/// Demonstrează stabilirea conexiunii cu portofelul blockchain
/// și executarea operațiunilor de bază
/// 
public class WalletConnection : MonoBehaviour
{
    [Header("Configurația blockchain")]
    [SerializeField] private string rpcUrl = "https://mainnet.infura.io/v3/YOUR_PROJECT_ID";
    
    private Web3 web3;
    public string PlayerWalletAddress { get; private set; }
    
    // Evenimente pentru actualizarea UI
    public event Action<string> OnWalletConnected;
    public event Action<decimal> OnBalanceUpdated;

    void Start()
    {
        InitializeWeb3();
    }

    /// 
    /// Inițializarea conexiunii Web3
    /// 
    private void InitializeWeb3()
    {
        try
        {
            web3 = new Web3(rpcUrl);
            Debug.Log("Conexiunea Web3 inițializată cu succes");
        }
        catch (Exception ex)
        {
            Debug.LogError($"Eroare la inițializarea Web3: {ex.Message}");
        }
    }

    /// 
    /// Conectarea la portofelul MetaMask
    /// 
    public async Task<bool> ConnectWallet()
    {
        try
        {
            // Solicitarea accesului la cont
            var accounts = await RequestAccountAccess();
            
            if (accounts != null && accounts.Length > 0)
            {
                PlayerWalletAddress = accounts[0];
                Debug.Log($"Portofel conectat: {PlayerWalletAddress}");
                
                OnWalletConnected?.Invoke(PlayerWalletAddress);
                await UpdateBalance();
                
                return true;
            }
            
            return false;
        }
        catch (Exception ex)
        {
            Debug.LogError($"Eroare la conectarea portofelului: {ex.Message}");
            return false;
        }
    }

    /// 
    /// Obținerea soldului curent ETH
    /// 
    public async Task UpdateBalance()
    {
        if (string.IsNullOrEmpty(PlayerWalletAddress)) return;

        try
        {
            var balanceWei = await web3.Eth.GetBalance.SendRequestAsync(PlayerWalletAddress);
            var balanceEth = Web3.Convert.FromWei(balanceWei.Value);
            
            Debug.Log($"Soldul portofelului: {balanceEth} ETH");
            OnBalanceUpdated?.Invoke(balanceEth);
        }
        catch (Exception ex)
        {
            Debug.LogError($"Eroare la obținerea soldului: {ex.Message}");
        }
    }
}

Exemplu: Gestionarea NFT

Sistem pentru lucrul cu tokenurile NFT în joc:

using System.Collections.Generic;
using UnityEngine;

/// 
/// Gestionarea NFT în jocul Unity
/// Încărcarea, afișarea și interacțiunea cu tokenurile NFT
/// 
public class NFTManager : MonoBehaviour 
{
    [System.Serializable]
    public class GameNFT
    {
        public string tokenId;
        public string name;
        public string description;
        public string imageUrl;
        public string rarity;
        public Dictionary<string, object> attributes;
        
        public GameNFT(string id, string nftName, string desc, string image, string nftRarity)
        {
            tokenId = id;
            name = nftName;
            description = desc;
            imageUrl = image;
            rarity = nftRarity;
            attributes = new Dictionary<string, object>();
        }
    }

    private List<GameNFT> playerNFTs = new List<GameNFT>();
    
    /// 
    /// Încărcarea NFT-urilor jucătorului din blockchain
    /// 
    public async Task LoadPlayerNFTs(string playerAddress)
    {
        try
        {
            playerNFTs.Clear();
            
            // Obținerea numărului de NFT-uri ale jucătorului
            var balance = await GetNFTBalance(playerAddress);
            Debug.Log($"Jucătorul deține {balance} NFT-uri");
            
            // Încărcarea fiecărui NFT
            for (int i = 0; i < balance; i++)
            {
                var tokenId = await GetTokenByIndex(playerAddress, i);
                var nft = await LoadNFTMetadata(tokenId);
                
                if (nft != null)
                {
                    playerNFTs.Add(nft);
                }
            }
            
            Debug.Log($"Încărcate {playerNFTs.Count} NFT-uri pentru jucător");
        }
        catch (Exception ex)
        {
            Debug.LogError($"Eroare la încărcarea NFT-urilor: {ex.Message}");
        }
    }

    /// 
    /// Filtrarea NFT-urilor după raritate
    /// 
    public List<GameNFT> GetNFTsByRarity(string rarity)
    {
        return playerNFTs.FindAll(nft => 
            nft.rarity.Equals(rarity, StringComparison.OrdinalIgnoreCase));
    }
}

Exemplu: Contract smart Solidity

Contract smart de bază pentru tokenurile de joc:

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

/**
 * @title GameCoin
 * @dev Moneda de joc pentru jocul Unity
 * Include mecanici play-to-earn și achiziții din joc
 */
contract GameCoin is ERC20, Ownable {
    uint256 public constant MAX_SUPPLY = 1000000000 * 10**18; // 1 miliard de tokeni
    
    // Mapping pentru urmărirea realizărilor jucătorilor
    mapping(address => uint256) public playerScore;
    mapping(address => uint256) public lastRewardClaim;
    
    // Evenimente pentru integrarea cu jocul
    event RewardClaimed(address indexed player, uint256 amount, uint256 score);
    event ScoreUpdated(address indexed player, uint256 newScore);
    
    constructor() ERC20("GameCoin", "GC") {
        // Emisia inițială pentru proprietarul contractului
        _mint(msg.sender, 100000000 * 10**18); // 100 milioane tokeni
    }
    
    /**
     * @dev Actualizarea scorului jucătorului (apelată de serverul de joc)
     * Doar proprietarul contractului poate actualiza scorurile
     */
    function updatePlayerScore(address player, uint256 score) external onlyOwner {
        playerScore[player] = score;
        emit ScoreUpdated(player, score);
    }
    
    /**
     * @dev Revendicarea recompensei zilnice de către jucător
     * Jucătorii pot revendica recompense o dată pe zi pe baza scorului lor
     */
    function claimDailyReward() external {
        require(block.timestamp >= lastRewardClaim[msg.sender] + 1 days, 
                "Recompensa deja revendicată astăzi");
        require(playerScore[msg.sender] > 0, "Niciun scor înregistrat");
        
        // Calcularea recompensei pe baza scorului jucătorului (1 token la 100 puncte)
        uint256 reward = playerScore[msg.sender] / 100;
        require(reward > 0, "Scor prea mic pentru recompensă");
        
        // Verificarea ofertei maxime
        require(totalSupply() + reward <= MAX_SUPPLY, 
                "Depășirea ofertei maxime");
        
        // Actualizarea timpului ultimei revendicări
        lastRewardClaim[msg.sender] = block.timestamp;
        
        // Emiterea tokenurilor de recompensă către jucător
        _mint(msg.sender, reward);
        
        emit RewardClaimed(msg.sender, reward, playerScore[msg.sender]);
    }
    
    /**
     * @dev Arderea tokenurilor (pentru achizițiile din joc)
     */
    function burn(uint256 amount) external {
        _burn(msg.sender, amount);
    }
}
