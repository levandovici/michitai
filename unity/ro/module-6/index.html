Modulul 6: Unity Audio și Sistem Sonor

Exemple cuprinzătoare pentru sistemul audio Unity și gestionarea sunetului

Bazele Sistemului Audio

Gestionarea de bază a audio și configurarea sistemului sonor Unity.

using UnityEngine; public class AudioManager : MonoBehaviour { [Header("Clipuri Audio")] public AudioClip backgroundMusic; public AudioClip buttonClickSound; public AudioClip explosionSound; [Header("Surse Audio")] public AudioSource musicSource; public AudioSource sfxSource; [Header("Setări Volum")] [Range(0f, 1f)] public float masterVolume = 1f; [Range(0f, 1f)] public float musicVolume = 0.7f; [Range(0f, 1f)] public float sfxVolume = 1f; void Start() { SetupAudioSources(); PlayBackgroundMusic(); Debug.Log("Audio Manager inițializat"); } void SetupAudioSources() { if (musicSource == null) { musicSource = gameObject.AddComponent<AudioSource>(); } musicSource.loop = true; musicSource.playOnAwake = false; musicSource.volume = musicVolume; if (sfxSource == null) { sfxSource = gameObject.AddComponent<AudioSource>(); } sfxSource.loop = false; sfxSource.playOnAwake = false; sfxSource.volume = sfxVolume; Debug.Log("Sursele audio configurate"); } public void PlayBackgroundMusic() { if (backgroundMusic != null && musicSource != null) { musicSource.clip = backgroundMusic; musicSource.Play(); Debug.Log("Muzica de fundal pornită"); } } public void PlaySoundEffect(AudioClip clip) { if (clip != null && sfxSource != null) { sfxSource.PlayOneShot(clip); Debug.Log("Efect sonor redat: " + clip.name); } } public void SetMasterVolume(float volume) { masterVolume = Mathf.Clamp01(volume); AudioListener.volume = masterVolume; Debug.Log("Volumul general: " + masterVolume); } public void SetMusicVolume(float volume) { musicVolume = Mathf.Clamp01(volume); if (musicSource != null) { musicSource.volume = musicVolume; } Debug.Log("Volumul muzicii: " + musicVolume); } }

Gestionarea AudioSource

Lucrul detaliat cu componenta AudioSource și parametrii săi.

using UnityEngine; public class AudioSourceController : MonoBehaviour { [Header("Componente AudioSource")] public AudioSource primarySource; [Header("Clipuri Audio")] public AudioClip[] musicTracks; [Header("Parametri Redare")] [Range(0f, 3f)] public float pitch = 1f; [Range(0f, 1f)] public float volume = 1f; [Range(-1f, 1f)] public float stereoPan = 0f; private int currentTrackIndex = 0; private bool isPlaying = false; void Start() { if (primarySource == null) primarySource = GetComponent<AudioSource>(); Debug.Log("AudioSource Controller inițializat"); } void Update() { UpdateAudioParameters(); CheckPlaybackStatus(); HandleInput(); } void UpdateAudioParameters() { if (primarySource != null) { primarySource.pitch = pitch; primarySource.volume = volume; primarySource.panStereo = stereoPan; } } void CheckPlaybackStatus() { if (primarySource != null && isPlaying && !primarySource.isPlaying) { Debug.Log("Redarea completă"); isPlaying = false; if (musicTracks.Length > 0) { PlayNextTrack(); } } } void HandleInput() { if (Input.GetKeyDown(KeyCode.Space)) { TogglePlayback(); } if (Input.GetKeyDown(KeyCode.N)) { PlayNextTrack(); } if (Input.GetKeyDown(KeyCode.P)) { PlayPreviousTrack(); } } public void TogglePlayback() { if (primarySource == null) return; if (primarySource.isPlaying) { primarySource.Pause(); Debug.Log("Redare întreruptă"); } else { primarySource.UnPause(); isPlaying = true; Debug.Log("Redare reluată"); } } public void PlayNextTrack() { if (musicTracks.Length > 0) { currentTrackIndex = (currentTrackIndex + 1) % musicTracks.Length; PlayCurrentTrack(); Debug.Log($"Următoarea piesă: {currentTrackIndex + 1}/{musicTracks.Length}"); } } public void PlayPreviousTrack() { if (musicTracks.Length > 0) { currentTrackIndex = (currentTrackIndex - 1 + musicTracks.Length) % musicTracks.Length; PlayCurrentTrack(); Debug.Log($"Piesa anterioară: {currentTrackIndex + 1}/{musicTracks.Length}"); } } public void PlayCurrentTrack() { if (musicTracks.Length > 0 && primarySource != null) { AudioClip currentClip = musicTracks[currentTrackIndex]; primarySource.clip = currentClip; primarySource.Play(); isPlaying = true; Debug.Log("Se redă: " + currentClip.name); } } }

Audio Mixer și Grupuri

Lucrul cu Audio Mixer pentru mixaj profesional de sunet.

using UnityEngine; using UnityEngine.Audio; public class AudioMixerController : MonoBehaviour { [Header("Audio Mixer")] public AudioMixer mainMixer; [Header("Parametri Mixer")] [Range(-80f, 20f)] public float masterVolume = 0f; [Range(-80f, 20f)] public float musicVolume = 0f; [Range(-80f, 20f)] public float sfxVolume = 0f; void Start() { SetupMixer(); Debug.Log("Audio Mixer Controller inițializat"); } void SetupMixer() { if (mainMixer != null) { SetMasterVolume(masterVolume); SetMusicVolume(musicVolume); SetSFXVolume(sfxVolume); Debug.Log("Audio Mixer configurat"); } } public void SetMasterVolume(float volume) { masterVolume = volume; if (mainMixer != null) { mainMixer.SetFloat("MasterVolume", volume); Debug.Log("Volum Master: " + volume + " dB"); } } public void SetMusicVolume(float volume) { musicVolume = volume; if (mainMixer != null) { mainMixer.SetFloat("MusicVolume", volume); Debug.Log("Volum Muzică: " + volume + " dB"); } } public void SetSFXVolume(float volume) { sfxVolume = volume; if (mainMixer != null) { mainMixer.SetFloat("SFXVolume", volume); Debug.Log("Volum SFX: " + volume + " dB"); } } public void SetLowPassFilter(float frequency) { if (mainMixer != null) { mainMixer.SetFloat("LowPassFreq", frequency); Debug.Log("Filtru Low Pass: " + frequency + " Hz"); } } public void SetReverb(float level) { if (mainMixer != null) { mainMixer.SetFloat("ReverbLevel", level); Debug.Log("Nivel Reverb: " + level); } } public void CreateSnapshot(string snapshotName) { if (mainMixer != null) { AudioMixerSnapshot snapshot = mainMixer.FindSnapshot(snapshotName); if (snapshot != null) { snapshot.TransitionTo(1f); Debug.Log("Tranziție la snapshot: " + snapshotName); } } } void Update() { if (Input.GetKeyDown(KeyCode.Alpha1)) { CreateSnapshot("Normal"); } if (Input.GetKeyDown(KeyCode.Alpha2)) { CreateSnapshot("Underwater"); } if (Input.GetKeyDown(KeyCode.Alpha3)) { CreateSnapshot("Cave"); } } }

Sunet 3D Spațial

Implementarea audio 3D cu poziționare spațială.

using UnityEngine; public class SpatialAudioController : MonoBehaviour { [Header("Setări Audio 3D")] public AudioSource spatialSource; [Header("Parametri Spațiali")] [Range(0f, 1f)] public float spatialBlend = 1f; [Range(0f, 5f)] public float dopplerLevel = 1f; [Range(0f, 360f)] public float spread = 0f; [Header("Setări Distanță")] public float minDistance = 1f; public float maxDistance = 500f; public AudioRolloffMode rolloffMode = AudioRolloffMode.Logarithmic; void Start() { SetupSpatialAudio(); Debug.Log("Spatial Audio Controller inițializat"); } void SetupSpatialAudio() { if (spatialSource == null) spatialSource = GetComponent<AudioSource>(); if (spatialSource != null) { spatialSource.spatialBlend = spatialBlend; spatialSource.dopplerLevel = dopplerLevel; spatialSource.spread = spread; spatialSource.rolloffMode = rolloffMode; spatialSource.minDistance = minDistance; spatialSource.maxDistance = maxDistance; Debug.Log("Audio 3D configurat"); } } void Update() { UpdateSpatialParameters(); HandleInput(); } void UpdateSpatialParameters() { if (spatialSource != null) { spatialSource.spatialBlend = spatialBlend; spatialSource.dopplerLevel = dopplerLevel; spatialSource.spread = spread; spatialSource.minDistance = minDistance; spatialSource.maxDistance = maxDistance; spatialSource.rolloffMode = rolloffMode; } } void HandleInput() { if (Input.GetKeyDown(KeyCode.Plus)) { IncreaseDistance(); } if (Input.GetKeyDown(KeyCode.Minus)) { DecreaseDistance(); } if (Input.GetKeyDown(KeyCode.D)) { ToggleDoppler(); } } public void IncreaseDistance() { maxDistance = Mathf.Min(maxDistance + 50f, 1000f); Debug.Log("Distanța maximă: " + maxDistance); } public void DecreaseDistance() { maxDistance = Mathf.Max(maxDistance - 50f, minDistance + 1f); Debug.Log("Distanța maximă: " + maxDistance); } public void ToggleDoppler() { dopplerLevel = dopplerLevel > 0 ? 0f : 1f; Debug.Log("Efectul Doppler: " + (dopplerLevel > 0 ? "activat" : "dezactivat")); } public void SetSpatialBlend(float blend) { spatialBlend = Mathf.Clamp01(blend); Debug.Log("Amestec spațial: " + spatialBlend); } }

Tehnici Audio Avansate

Muzică dinamică, efecte audio și optimizarea performanței.

using UnityEngine; using System.Collections; public class AdvancedAudioController : MonoBehaviour { [Header("Muzică Dinamică")] public AudioSource[] musicLayers; public float crossfadeDuration = 2f; [Header("Efecte Audio")] public AudioLowPassFilter lowPassFilter; public AudioHighPassFilter highPassFilter; public AudioReverbFilter reverbFilter; private int currentMusicLayer = 0; private bool isCrossfading = false; void Start() { SetupAdvancedAudio(); Debug.Log("Advanced Audio Controller inițializat"); } void SetupAdvancedAudio() { for (int i = 0; i < musicLayers.Length; i++) { if (musicLayers[i] != null) { musicLayers[i].volume = (i == 0) ? 1f : 0f; musicLayers[i].Play(); } } if (lowPassFilter == null) lowPassFilter = gameObject.AddComponent<AudioLowPassFilter>(); if (highPassFilter == null) highPassFilter = gameObject.AddComponent<AudioHighPassFilter>(); if (reverbFilter == null) reverbFilter = gameObject.AddComponent<AudioReverbFilter>(); } public void CrossfadeToLayer(int layerIndex) { if (layerIndex >= 0 && layerIndex < musicLayers.Length && !isCrossfading) { StartCoroutine(CrossfadeCoroutine(layerIndex)); } } IEnumerator CrossfadeCoroutine(int targetLayer) { isCrossfading = true; int previousLayer = currentMusicLayer; currentMusicLayer = targetLayer; float elapsedTime = 0f; while (elapsedTime < crossfadeDuration) { elapsedTime += Time.deltaTime; float t = elapsedTime / crossfadeDuration; if (musicLayers[previousLayer] != null) musicLayers[previousLayer].volume = Mathf.Lerp(1f, 0f, t); if (musicLayers[targetLayer] != null) musicLayers[targetLayer].volume = Mathf.Lerp(0f, 1f, t); yield return null; } if (musicLayers[previousLayer] != null) musicLayers[previousLayer].volume = 0f; if (musicLayers[targetLayer] != null) musicLayers[targetLayer].volume = 1f; isCrossfading = false; Debug.Log($"Comutare la stratul {targetLayer} completă"); } public void ApplyLowPassFilter(float frequency) { if (lowPassFilter != null) { lowPassFilter.cutoffFrequency = Mathf.Clamp(frequency, 10f, 22000f); Debug.Log("Filtru Low Pass: " + frequency + " Hz"); } } public void ApplyReverb(AudioReverbPreset preset) { if (reverbFilter != null) { reverbFilter.reverbPreset = preset; Debug.Log("Preset Reverb: " + preset); } } void Update() { if (Input.GetKeyDown(KeyCode.Alpha1)) { CrossfadeToLayer(0); } if (Input.GetKeyDown(KeyCode.Alpha2)) { CrossfadeToLayer(1); } if (Input.GetKeyDown(KeyCode.L)) { ApplyLowPassFilter(1000f); } if (Input.GetKeyDown(KeyCode.R)) { ApplyReverb(AudioReverbPreset.Cave); } } }
