Modulul 10: Optimizare Avansată și Lansare

Exemple complete de gestionare a memoriei, dezvoltare cross-platform și monetizare

Gestionarea Memoriei cu Addressables

Încărcarea și descărcarea eficientă a resurselor pentru optimizarea memoriei.

using UnityEngine; using UnityEngine.AddressableAssets; using UnityEngine.ResourceManagement.AsyncOperations; using System.Collections.Generic; public class AddressableManager : MonoBehaviour { public AssetReference playerPrefab; public AssetReference audioClip; private Dictionary<string, AsyncOperationHandle> loadedAssets = new Dictionary<string, AsyncOperationHandle>(); void Start() { LoadAssets(); Debug.Log("Addressable Manager inițializat"); } async void LoadAssets() { // Încărcarea prefab-ului jucătorului var playerHandle = playerPrefab.LoadAssetAsync<GameObject>(); await playerHandle.Task; if (playerHandle.Status == AsyncOperationStatus.Succeeded) { loadedAssets["player"] = playerHandle; var player = Instantiate(playerHandle.Result); Debug.Log("Jucător creat din resursa Addressable"); } // Încărcarea audio var audioHandle = audioClip.LoadAssetAsync<AudioClip>(); await audioHandle.Task; if (audioHandle.Status == AsyncOperationStatus.Succeeded) { loadedAssets["audio"] = audioHandle; PlayAudio(audioHandle.Result); } } void PlayAudio(AudioClip clip) { var audioSource = GetComponent<AudioSource>(); if (audioSource == null) audioSource = gameObject.AddComponent<AudioSource>(); audioSource.clip = clip; audioSource.Play(); Debug.Log("Audio redă"); } public void UnloadAsset(string key) { if (loadedAssets.ContainsKey(key)) { Addressables.Release(loadedAssets[key]); loadedAssets.Remove(key); Debug.Log($"Resursa {key} descărcată"); } } void OnDestroy() { foreach (var asset in loadedAssets.Values) { Addressables.Release(asset); } Debug.Log("Toate resursele Addressable eliberate"); } }

Dezvoltare Cross-Platform

Adaptarea jocului pentru diferite platforme: Android, WebGL, PC.

using UnityEngine; public class PlatformManager : MonoBehaviour { public bool enableTouchControls = false; public float mobileUIScale = 1.2f; public float desktopUIScale = 1.0f; void Start() { DetectAndConfigurePlatform(); } void DetectAndConfigurePlatform() { switch (Application.platform) { case RuntimePlatform.Android: case RuntimePlatform.IPhonePlayer: SetupMobile(); break; case RuntimePlatform.WebGLPlayer: SetupWebGL(); break; default: SetupDesktop(); break; } Debug.Log($"Platformă: {Application.platform}"); } void SetupMobile() { enableTouchControls = true; SetUIScale(mobileUIScale); QualitySettings.SetQualityLevel(1); // Calitate scăzută // Activarea elementelor tactile var touchControls = GameObject.FindGameObjectsWithTag("TouchControl"); foreach (var control in touchControls) { control.SetActive(true); } Debug.Log("Configurație mobilă aplicată"); } void SetupWebGL() { enableTouchControls = SystemInfo.deviceType == DeviceType.Handheld; SetUIScale(enableTouchControls ? mobileUIScale : desktopUIScale); QualitySettings.SetQualityLevel(2); // Calitate medie Debug.Log("Configurație WebGL aplicată"); } void SetupDesktop() { enableTouchControls = false; SetUIScale(desktopUIScale); QualitySettings.SetQualityLevel(4); // Calitate înaltă // Dezactivarea elementelor tactile var touchControls = GameObject.FindGameObjectsWithTag("TouchControl"); foreach (var control in touchControls) { control.SetActive(false); } Debug.Log("Configurație desktop aplicată"); } void SetUIScale(float scale) { var canvas = FindObjectOfType<Canvas>(); if (canvas != null) { var scaler = canvas.GetComponent<UnityEngine.UI.CanvasScaler>(); if (scaler != null) { scaler.scaleFactor = scale; } } } }

Analiză și Urmărire

Colectarea datelor despre comportamentul jucătorilor și performanța jocului.

using UnityEngine; using System.Collections.Generic; public class AnalyticsManager : MonoBehaviour { public string gameVersion = "1.0.0"; private float sessionStartTime; private int levelCompletions = 0; void Start() { InitializeAnalytics(); TrackSessionStart(); } void InitializeAnalytics() { sessionStartTime = Time.time; Debug.Log("Analiză inițializată"); } void TrackSessionStart() { var data = new Dictionary<string, object>(); data["device"] = SystemInfo.deviceModel; data["os"] = SystemInfo.operatingSystem; data["version"] = gameVersion; data["platform"] = Application.platform.ToString(); TrackEvent("session_start", data); } public void TrackEvent(string eventName, Dictionary<string, object> parameters = null) { var eventData = new Dictionary<string, object>(); eventData["event"] = eventName; eventData["time"] = System.DateTime.Now.ToString(); eventData["session_time"] = Time.time - sessionStartTime; if (parameters != null) { foreach (var param in parameters) { eventData[param.Key] = param.Value; } } Debug.Log($"Eveniment: {eventName}"); LogEvent(eventData); } public void TrackLevelComplete(int level, float time) { levelCompletions++; var data = new Dictionary<string, object>(); data["level"] = level; data["completion_time"] = time; data["total_completions"] = levelCompletions; TrackEvent("level_complete", data); } public void TrackPurchase(string item, float price) { var data = new Dictionary<string, object>(); data["item"] = item; data["price"] = price; TrackEvent("purchase", data); } void LogEvent(Dictionary<string, object> data) { string log = "Analiză:\n"; foreach (var item in data) { log += $" {item.Key}: {item.Value}\n"; } Debug.Log(log); } void OnDestroy() { var sessionTime = Time.time - sessionStartTime; var data = new Dictionary<string, object>(); data["duration"] = sessionTime; data["completions"] = levelCompletions; TrackEvent("session_end", data); } }

Monetizare cu Unity Ads

Integrarea reclamelor și a achizițiilor în joc pentru monetizare.

using UnityEngine; using UnityEngine.UI; public class MonetizationManager : MonoBehaviour { [Header("Setări")] public string gameId = "1234567"; public Button watchAdButton; public Text coinsText; private int playerCoins = 0; private bool adReady = false; void Start() { InitializeMonetization(); LoadPlayerData(); } void InitializeMonetization() { // Simularea inițializării Unity Ads Invoke("SetAdReady", 2f); if (watchAdButton != null) { watchAdButton.onClick.AddListener(ShowRewardedAd); } Debug.Log("Monetizare inițializată"); } void SetAdReady() { adReady = true; UpdateUI(); Debug.Log("Reclamă pregătită"); } void LoadPlayerData() { playerCoins = PlayerPrefs.GetInt("Coins", 0); UpdateUI(); } void SavePlayerData() { PlayerPrefs.SetInt("Coins", playerCoins); PlayerPrefs.Save(); } public void ShowRewardedAd() { if (!adReady) return; Debug.Log("Afișare reclamă..."); adReady = false; UpdateUI(); // Simularea vizionării reclamei Invoke("OnAdComplete", 3f); } void OnAdComplete() { int reward = 50; playerCoins += reward; SavePlayerData(); UpdateUI(); Debug.Log($"Primit {reward} monede pentru reclamă"); // Următoarea reclamă în 30 de secunde Invoke("SetAdReady", 30f); } public void PurchaseCoins(int amount, float price) { playerCoins += amount; SavePlayerData(); UpdateUI(); // Trimiterea evenimentului în analiză var analytics = FindObjectOfType<AnalyticsManager>(); if (analytics != null) { analytics.TrackPurchase($"coins_{amount}", price); } Debug.Log($"Cumpărat {amount} monede pentru ${price}"); } public bool SpendCoins(int amount) { if (playerCoins >= amount) { playerCoins -= amount; SavePlayerData(); UpdateUI(); Debug.Log($"Cheltuit {amount} monede"); return true; } Debug.Log("Monede insuficiente"); return false; } void UpdateUI() { if (coinsText != null) { coinsText.text = $"Monede: {playerCoins}"; } if (watchAdButton != null) { watchAdButton.interactable = adReady; } } }

Pregătirea pentru Publicare

Automatizarea procesului de build și pregătirea jocului pentru lansare.

using UnityEngine; #if UNITY_EDITOR using UnityEditor; using UnityEditor.Build.Reporting; #endif public class BuildManager : MonoBehaviour { [Header("Setări Build")] public string productName = "MyGame"; public string version = "1.0.0"; #if UNITY_EDITOR [ContextMenu("Build Android")] public void BuildAndroid() { SetupBuildSettings(); var buildOptions = new BuildPlayerOptions(); buildOptions.scenes = GetScenePaths(); buildOptions.locationPathName = $"Builds/Android/{productName}.apk"; buildOptions.target = BuildTarget.Android; buildOptions.options = BuildOptions.None; Debug.Log("Începerea build-ului pentru Android..."); var report = BuildPipeline.BuildPlayer(buildOptions); if (report.summary.result == BuildResult.Succeeded) { Debug.Log("Build Android finalizat cu succes!"); } else { Debug.LogError("Eroare build Android"); } } [ContextMenu("Build Windows")] public void BuildWindows() { SetupBuildSettings(); var buildOptions = new BuildPlayerOptions(); buildOptions.scenes = GetScenePaths(); buildOptions.locationPathName = $"Builds/Windows/{productName}.exe"; buildOptions.target = BuildTarget.StandaloneWindows64; buildOptions.options = BuildOptions.None; Debug.Log("Începerea build-ului pentru Windows..."); var report = BuildPipeline.BuildPlayer(buildOptions); if (report.summary.result == BuildResult.Succeeded) { Debug.Log("Build Windows finalizat cu succes!"); } } void SetupBuildSettings() { PlayerSettings.productName = productName; PlayerSettings.bundleVersion = version; // Setări Android PlayerSettings.Android.bundleVersionCode++; PlayerSettings.Android.minSdkVersion = AndroidSdkVersions.AndroidApiLevel21; Debug.Log($"Setări build: {productName} v{version}"); } string[] GetScenePaths() { var scenes = new string[EditorBuildSettings.scenes.Length]; for (int i = 0; i < scenes.Length; i++) { scenes[i] = EditorBuildSettings.scenes[i].path; } return scenes; } [ContextMenu("Pregătire pentru Magazin")] public void PrepareForStore() { // Setarea configurațiilor pentru magazin PlayerSettings.Android.useAPKExpansionFiles = false; PlayerSettings.Android.keystoreName = "user.keystore"; Debug.Log("Jocul pregătit pentru publicare în magazin"); } #endif }
