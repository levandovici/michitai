Modulul 5: Unity UI și Canvas

Exemple cuprinzătoare pentru sistemul de interfață utilizator Unity

Bazele lucrului cu Canvas

Gestionarea Canvas și setările de bază ale sistemului UI.

using UnityEngine; using UnityEngine.UI; public class CanvasManager : MonoBehaviour { [Header("Setări Canvas")] public Canvas mainCanvas; public CanvasScaler canvasScaler; void Start() { if (mainCanvas == null) mainCanvas = GetComponent<Canvas>(); Debug.Log("Canvas Manager inițializat"); SetupCanvas(); } void SetupCanvas() { mainCanvas.renderMode = RenderMode.ScreenSpaceOverlay; mainCanvas.sortingOrder = 0; if (canvasScaler != null) { canvasScaler.uiScaleMode = CanvasScaler.ScaleMode.ScaleWithScreenSize; canvasScaler.referenceResolution = new Vector2(1920, 1080); canvasScaler.matchWidthOrHeight = 0.5f; } Debug.Log("Canvas configurat pentru rezoluția: 1920x1080"); } public void SetRenderMode(RenderMode mode) { mainCanvas.renderMode = mode; Debug.Log("Modul de randare schimbat la: " + mode); } public void ToggleCanvas() { mainCanvas.enabled = !mainCanvas.enabled; Debug.Log("Canvas " + (mainCanvas.enabled ? "activat" : "dezactivat")); } }

Gestionarea Elementelor UI

Lucrul cu componentele UI de bază: butoane, text, imagini, slidere.

using UnityEngine; using UnityEngine.UI; using TMPro; public class UIElementsController : MonoBehaviour { [Header("Elemente UI")] public Button mainButton; public TextMeshProUGUI displayText; public Image statusImage; public Slider progressSlider; public Toggle settingsToggle; private int clickCounter = 0; void Start() { SetupUIElements(); Debug.Log("UI Elements Controller inițializat"); } void SetupUIElements() { if (mainButton != null) { mainButton.onClick.AddListener(OnMainButtonClick); } if (progressSlider != null) { progressSlider.onValueChanged.AddListener(OnSliderValueChanged); progressSlider.minValue = 0f; progressSlider.maxValue = 100f; } if (settingsToggle != null) { settingsToggle.onValueChanged.AddListener(OnToggleChanged); } UpdateDisplay(); } public void OnMainButtonClick() { clickCounter++; Debug.Log("Buton apăsat! Contor: " + clickCounter); if (progressSlider != null) { progressSlider.value = Mathf.Min(progressSlider.value + 10f, 100f); } UpdateDisplay(); } public void OnSliderValueChanged(float value) { Debug.Log("Valoarea sliderului: " + value.ToString("F1")); if (statusImage != null) { Color newColor = Color.Lerp(Color.red, Color.green, value / 100f); statusImage.color = newColor; } UpdateDisplay(); } public void OnToggleChanged(bool isOn) { Debug.Log("Comutator: " + (isOn ? "Activat" : "Dezactivat")); if (mainButton != null) { mainButton.interactable = isOn; } UpdateDisplay(); } void UpdateDisplay() { if (displayText != null) { string info = $"Clicuri: {clickCounter}\n"; info += $"Progres: {(progressSlider != null ? progressSlider.value.ToString("F1") : "0")}%\n"; info += $"Setări: {(settingsToggle != null ? (settingsToggle.isOn ? "PORNIT" : "OPRIT") : "NICIUNUL")}"; displayText.text = info; } } }

Interacțiuni și Evenimente UI

Gestionarea evenimentelor UI, animații și elemente interactive.

using UnityEngine; using UnityEngine.UI; using UnityEngine.EventSystems; using TMPro; public class UIInteractionManager : MonoBehaviour, IPointerEnterHandler, IPointerExitHandler { [Header("Elemente Interactive")] public Button[] menuButtons; public TextMeshProUGUI feedbackText; [Header("Setări Animație")] public float hoverScale = 1.1f; public float animationDuration = 0.3f; private Vector3 originalScale; void Start() { SetupInteractions(); originalScale = transform.localScale; Debug.Log("UI Interaction Manager inițializat"); } void SetupInteractions() { for (int i = 0; i < menuButtons.Length; i++) { int buttonIndex = i; if (menuButtons[i] != null) { menuButtons[i].onClick.AddListener(() => OnMenuButtonClick(buttonIndex)); } } UpdateFeedback("Sistemul de interacțiuni este gata"); } public void OnMenuButtonClick(int buttonIndex) { Debug.Log($"Butonul de meniu {buttonIndex} apăsat"); string[] buttonNames = {"Acasă", "Setări", "Profil", "Ieșire"}; string buttonName = buttonIndex < buttonNames.Length ? buttonNames[buttonIndex] : "Necunoscut"; UpdateFeedback($"Selectat: {buttonName}"); StartCoroutine(AnimateFeedback()); } public void OnPointerEnter(PointerEventData eventData) { Debug.Log("Cursorul a intrat pe elementul UI"); UpdateFeedback("Cursor introdus"); StartCoroutine(ScaleTo(originalScale * hoverScale)); } public void OnPointerExit(PointerEventData eventData) { Debug.Log("Cursorul a părăsit elementul UI"); UpdateFeedback("Cursor ieșit"); StartCoroutine(ScaleTo(originalScale)); } void UpdateFeedback(string message) { if (feedbackText != null) { feedbackText.text = $"[{System.DateTime.Now:HH:mm:ss}] {message}"; } } System.Collections.IEnumerator AnimateFeedback() { if (feedbackText != null) { Color originalColor = feedbackText.color; for (int i = 0; i < 3; i++) { feedbackText.color = Color.yellow; yield return new WaitForSeconds(0.1f); feedbackText.color = originalColor; yield return new WaitForSeconds(0.1f); } } } System.Collections.IEnumerator ScaleTo(Vector3 targetScale) { Vector3 startScale = transform.localScale; float elapsedTime = 0f; while (elapsedTime < animationDuration) { elapsedTime += Time.deltaTime; float t = elapsedTime / animationDuration; transform.localScale = Vector3.Lerp(startScale, targetScale, t); yield return null; } transform.localScale = targetScale; } }

Layout Groups și Aspect Adaptiv

Utilizarea Layout Groups pentru plasarea automată a elementelor UI.

using UnityEngine; using UnityEngine.UI; using TMPro; public class LayoutGroupManager : MonoBehaviour { [Header("Layout Groups")] public HorizontalLayoutGroup horizontalLayout; public VerticalLayoutGroup verticalLayout; public GridLayoutGroup gridLayout; [Header("Conținut Dinamic")] public GameObject buttonPrefab; public Transform dynamicContainer; public TextMeshProUGUI infoText; private int dynamicButtonCount = 0; void Start() { SetupLayoutGroups(); Debug.Log("Layout Group Manager inițializat"); } void SetupLayoutGroups() { if (horizontalLayout != null) { horizontalLayout.spacing = 10f; horizontalLayout.childControlWidth = true; horizontalLayout.childControlHeight = true; horizontalLayout.childAlignment = TextAnchor.MiddleCenter; } if (verticalLayout != null) { verticalLayout.spacing = 10f; verticalLayout.childControlWidth = true; verticalLayout.childForceExpandWidth = true; verticalLayout.childAlignment = TextAnchor.UpperCenter; } if (gridLayout != null) { gridLayout.cellSize = new Vector2(100, 100); gridLayout.spacing = new Vector2(10, 10); gridLayout.constraint = GridLayoutGroup.Constraint.FixedColumnCount; gridLayout.constraintCount = 3; } UpdateInfo(); } public void AddDynamicButton() { if (buttonPrefab != null && dynamicContainer != null) { GameObject newButton = Instantiate(buttonPrefab, dynamicContainer); dynamicButtonCount++; Button buttonComponent = newButton.GetComponent<Button>(); if (buttonComponent != null) { int buttonIndex = dynamicButtonCount; buttonComponent.onClick.AddListener(() => OnDynamicButtonClick(buttonIndex)); TextMeshProUGUI buttonText = newButton.GetComponentInChildren<TextMeshProUGUI>(); if (buttonText != null) { buttonText.text = $"Buton {buttonIndex}"; } } Debug.Log($"Buton dinamic {dynamicButtonCount} adăugat"); UpdateInfo(); LayoutRebuilder.ForceRebuildLayoutImmediate(dynamicContainer.GetComponent<RectTransform>()); } } public void RemoveDynamicButton() { if (dynamicContainer != null && dynamicContainer.childCount > 0) { Transform lastChild = dynamicContainer.GetChild(dynamicContainer.childCount - 1); DestroyImmediate(lastChild.gameObject); dynamicButtonCount = Mathf.Max(0, dynamicButtonCount - 1); Debug.Log("Ultimul buton dinamic eliminat"); UpdateInfo(); LayoutRebuilder.ForceRebuildLayoutImmediate(dynamicContainer.GetComponent<RectTransform>()); } } public void OnDynamicButtonClick(int buttonIndex) { Debug.Log($"Butonul dinamic {buttonIndex} apăsat"); UpdateInfo($"Ultima acțiune: Buton {buttonIndex}"); } void UpdateInfo(string additionalInfo = "") { if (infoText != null) { string info = $"Butoane dinamice: {dynamicButtonCount}"; if (!string.IsNullOrEmpty(additionalInfo)) { info += $"\n{additionalInfo}"; } infoText.text = info; } } }

Tehnici UI Avansate

Animații UI, măști, efecte și optimizarea performanței.

using UnityEngine; using UnityEngine.UI; using TMPro; public class AdvancedUIController : MonoBehaviour { [Header("Elemente Animație")] public RectTransform animatedPanel; public CanvasGroup fadeGroup; public Image progressBar; public TextMeshProUGUI animatedText; [Header("Setări Animație")] public float animationDuration = 1f; public AnimationCurve animationCurve = AnimationCurve.EaseInOut(0, 0, 1, 1); private bool isAnimating = false; void Start() { Debug.Log("Advanced UI Controller inițializat"); } public void AnimatePanel() { if (!isAnimating && animatedPanel != null) { StartCoroutine(AnimatePanelCoroutine()); } } System.Collections.IEnumerator AnimatePanelCoroutine() { isAnimating = true; Vector3 startPos = animatedPanel.anchoredPosition; Vector3 targetPos = startPos + Vector3.right * 200f; float elapsedTime = 0f; while (elapsedTime < animationDuration) { elapsedTime += Time.deltaTime; float t = elapsedTime / animationDuration; float curveValue = animationCurve.Evaluate(t); animatedPanel.anchoredPosition = Vector3.Lerp(startPos, targetPos, curveValue); yield return null; } animatedPanel.anchoredPosition = targetPos; yield return new WaitForSeconds(0.5f); elapsedTime = 0f; while (elapsedTime < animationDuration) { elapsedTime += Time.deltaTime; float t = elapsedTime / animationDuration; float curveValue = animationCurve.Evaluate(t); animatedPanel.anchoredPosition = Vector3.Lerp(targetPos, startPos, curveValue); yield return null; } animatedPanel.anchoredPosition = startPos; isAnimating = false; Debug.Log("Animația panoului completă"); } public void FadeInOut() { if (fadeGroup != null) { StartCoroutine(FadeCoroutine()); } } System.Collections.IEnumerator FadeCoroutine() { float startAlpha = fadeGroup.alpha; float elapsedTime = 0f; while (elapsedTime < animationDuration / 2) { elapsedTime += Time.deltaTime; float t = elapsedTime / (animationDuration / 2); fadeGroup.alpha = Mathf.Lerp(startAlpha, 0f, t); yield return null; } fadeGroup.alpha = 0f; elapsedTime = 0f; while (elapsedTime < animationDuration / 2) { elapsedTime += Time.deltaTime; float t = elapsedTime / (animationDuration / 2); fadeGroup.alpha = Mathf.Lerp(0f, 1f, t); yield return null; } fadeGroup.alpha = 1f; Debug.Log("Animația fade completă"); } public void AnimateProgressBar() { if (progressBar != null) { StartCoroutine(ProgressBarCoroutine()); } } System.Collections.IEnumerator ProgressBarCoroutine() { float elapsedTime = 0f; while (elapsedTime < animationDuration * 2) { elapsedTime += Time.deltaTime; float t = elapsedTime / (animationDuration * 2); progressBar.fillAmount = t; if (animatedText != null) { animatedText.text = $"Progres: {(t * 100):F0}%"; } yield return null; } progressBar.fillAmount = 1f; if (animatedText != null) { animatedText.text = "Progres: 100%"; } Debug.Log("Animația barei de progres completă"); } public void TypewriterEffect(string text) { if (animatedText != null) { StartCoroutine(TypewriterCoroutine(text)); } } System.Collections.IEnumerator TypewriterCoroutine(string text) { animatedText.text = ""; for (int i = 0; i <= text.Length; i++) { animatedText.text = text.Substring(0, i); yield return new WaitForSeconds(0.05f); } Debug.Log("Efectul de mașină de scris complet"); } void Update() { if (Input.GetKeyDown(KeyCode.Alpha1)) { AnimatePanel(); } if (Input.GetKeyDown(KeyCode.Alpha2)) { FadeInOut(); } if (Input.GetKeyDown(KeyCode.Alpha3)) { AnimateProgressBar(); } if (Input.GetKeyDown(KeyCode.Alpha4)) { TypewriterEffect("Acesta este un exemplu de efect de mașină de scris!"); } } }
