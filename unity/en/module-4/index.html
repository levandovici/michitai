Module 4: Unity Animation and Animator

Comprehensive examples for Unity animation system and Animator control

Basic Animator Control

Control Animator parameters and trigger animations from script.

using UnityEngine; public class AnimatorController : MonoBehaviour { private Animator animator; void Start() { animator = GetComponent<Animator>(); Debug.Log("Animator initialized for " + gameObject.name); } void Update() { if (Input.GetKeyDown(KeyCode.W)) { animator.SetBool("IsWalking", true); Debug.Log("Walking animation activated"); } if (Input.GetKeyUp(KeyCode.W)) { animator.SetBool("IsWalking", false); Debug.Log("Walking animation stopped"); } if (Input.GetKeyDown(KeyCode.Space)) { animator.SetTrigger("Jump"); Debug.Log("Jump animation triggered"); } } public void SetAnimationSpeed(float speed) { animator.SetFloat("Speed", speed); Debug.Log("Animation speed set to: " + speed); } }

Animation State Management

Manage transitions between states and monitor animation progress.

using UnityEngine; public class AnimationStateManager : MonoBehaviour { private Animator animator; private Rigidbody rb; [Header("Animation Parameters")] public float walkSpeed = 2f; public float runSpeed = 5f; private bool isGrounded = true; void Start() { animator = GetComponent<Animator>(); rb = GetComponent<Rigidbody>(); Debug.Log("Animation state manager initialized"); } void Update() { HandleMovementInput(); UpdateAnimationParameters(); MonitorAnimationStates(); } void HandleMovementInput() { float horizontal = Input.GetAxis("Horizontal"); float vertical = Input.GetAxis("Vertical"); Vector3 movement = new Vector3(horizontal, 0, vertical); bool isMoving = movement.magnitude > 0.1f; bool isRunning = Input.GetKey(KeyCode.LeftShift); animator.SetBool("IsWalking", isMoving && !isRunning); animator.SetBool("IsRunning", isMoving && isRunning); animator.SetFloat("MovementSpeed", movement.magnitude); } void UpdateAnimationParameters() { animator.SetBool("IsGrounded", isGrounded); animator.SetFloat("VerticalVelocity", rb.velocity.y); } void MonitorAnimationStates() { AnimatorStateInfo currentState = animator.GetCurrentAnimatorStateInfo(0); if (currentState.IsName("Jump") && currentState.normalizedTime >= 1.0f) { Debug.Log("Jump animation completed"); } } }

Blend Tree Control

Use Blend Trees for smooth and natural animations.

using UnityEngine; public class BlendTreeController : MonoBehaviour { private Animator animator; [Header("Blend Tree Settings")] public float accelerationTime = 2f; public float decelerationTime = 1f; private float currentBlendX = 0f; private float currentBlendY = 0f; private float targetBlendX = 0f; private float targetBlendY = 0f; void Start() { animator = GetComponent<Animator>(); Debug.Log("Blend Tree controller initialized"); } void Update() { HandleInput(); UpdateBlendValues(); SetAnimatorParameters(); } void HandleInput() { float inputX = Input.GetAxis("Horizontal"); float inputY = Input.GetAxis("Vertical"); targetBlendX = inputX; targetBlendY = inputY; if (Input.GetKey(KeyCode.LeftShift)) { targetBlendX *= 2f; targetBlendY *= 2f; } targetBlendX = Mathf.Clamp(targetBlendX, -1f, 1f); targetBlendY = Mathf.Clamp(targetBlendY, -1f, 1f); } void UpdateBlendValues() { float rate = 1f / accelerationTime; currentBlendX = Mathf.MoveTowards(currentBlendX, targetBlendX, rate * Time.deltaTime); currentBlendY = Mathf.MoveTowards(currentBlendY, targetBlendY, rate * Time.deltaTime); } void SetAnimatorParameters() { animator.SetFloat("BlendX", currentBlendX); animator.SetFloat("BlendY", currentBlendY); float magnitude = new Vector2(currentBlendX, currentBlendY).magnitude; animator.SetFloat("Speed", magnitude); animator.SetBool("IsMoving", magnitude > 0.1f); } }

Animation Events

Use Animation Events to synchronize effects with animations.

using UnityEngine; public class AnimationEventHandler : MonoBehaviour { [Header("Audio Components")] public AudioSource audioSource; public AudioClip footstepSound; public AudioClip jumpSound; [Header("Visual Effects")] public ParticleSystem dustEffect; void Start() { if (audioSource == null) audioSource = GetComponent<AudioSource>(); Debug.Log("Animation event handler initialized"); } // Method called by Animation Event for footsteps public void OnFootstep() { Debug.Log("Footstep event detected"); if (audioSource && footstepSound) { audioSource.PlayOneShot(footstepSound); } if (dustEffect) { dustEffect.Play(); } } // Method for jump start public void OnJumpStart() { Debug.Log("Jump start"); if (audioSource && jumpSound) { audioSource.PlayOneShot(jumpSound); } } // Method for landing public void OnLanding() { Debug.Log("Landing detected through event"); if (dustEffect) { dustEffect.Play(); } CameraShake(); } // Method for attack events public void OnAttackHit() { Debug.Log("Attack impact moment"); Collider[] enemies = Physics.OverlapSphere(transform.position, 2f); foreach (Collider enemy in enemies) { if (enemy.CompareTag("Enemy")) { Debug.Log("Enemy hit: " + enemy.name); } } } void CameraShake() { Camera mainCamera = Camera.main; if (mainCamera) { StartCoroutine(ShakeCoroutine(mainCamera)); } } System.Collections.IEnumerator ShakeCoroutine(Camera camera) { Vector3 originalPosition = camera.transform.position; float duration = 0.2f; float magnitude = 0.1f; float elapsed = 0f; while (elapsed < duration) { float x = Random.Range(-1f, 1f) * magnitude; float y = Random.Range(-1f, 1f) * magnitude; camera.transform.position = originalPosition + new Vector3(x, y, 0); elapsed += Time.deltaTime; yield return null; } camera.transform.position = originalPosition; } }

Advanced Animation Scripting

Advanced animation techniques including IK and procedural control.

using UnityEngine; public class AdvancedAnimationController : MonoBehaviour { private Animator animator; [Header("IK Settings")] public Transform leftHandTarget; public Transform rightHandTarget; public Transform lookAtTarget; [Range(0f, 1f)] public float leftHandWeight = 1f; [Range(0f, 1f)] public float rightHandWeight = 1f; [Range(0f, 1f)] public float lookAtWeight = 1f; void Start() { animator = GetComponent<Animator>(); Debug.Log("Advanced animation controller initialized"); } void Update() { HandleAdvancedControls(); } void HandleAdvancedControls() { if (Input.GetKey(KeyCode.Plus)) { animator.speed = Mathf.Min(animator.speed + Time.deltaTime, 3f); Debug.Log("Animation speed: " + animator.speed); } if (Input.GetKey(KeyCode.Minus)) { animator.speed = Mathf.Max(animator.speed - Time.deltaTime, 0.1f); Debug.Log("Animation speed: " + animator.speed); } if (Input.GetKeyDown(KeyCode.R)) { animator.speed = 1f; Debug.Log("Animation speed reset"); } } // IK callback - called automatically by Unity void OnAnimatorIK(int layerIndex) { if (animator) { // Set position and rotation for left hand if (leftHandTarget) { animator.SetIKPositionWeight(AvatarIKGoal.LeftHand, leftHandWeight); animator.SetIKRotationWeight(AvatarIKGoal.LeftHand, leftHandWeight); animator.SetIKPosition(AvatarIKGoal.LeftHand, leftHandTarget.position); animator.SetIKRotation(AvatarIKGoal.LeftHand, leftHandTarget.rotation); } // Set position and rotation for right hand if (rightHandTarget) { animator.SetIKPositionWeight(AvatarIKGoal.RightHand, rightHandWeight); animator.SetIKRotationWeight(AvatarIKGoal.RightHand, rightHandWeight); animator.SetIKPosition(AvatarIKGoal.RightHand, rightHandTarget.position); animator.SetIKRotation(AvatarIKGoal.RightHand, rightHandTarget.rotation); } // Set look at target if (lookAtTarget) { animator.SetLookAtWeight(lookAtWeight); animator.SetLookAtPosition(lookAtTarget.position); } } } public void SetLayerWeight(int layerIndex, float weight) { animator.SetLayerWeight(layerIndex, weight); Debug.Log($"Layer {layerIndex} weight set to {weight}"); } public void PlayAnimationAtTime(string stateName, float normalizedTime) { animator.Play(stateName, 0, normalizedTime); Debug.Log($"Animation {stateName} played at time {normalizedTime}"); } }
