Example 1: Variables and Data Types

Basic data types and working with variables in C#:

using UnityEngine;

public class VariablesExample : MonoBehaviour
{
    // Numeric types
    public int playerHealth = 100;
    public float moveSpeed = 5.5f;
    public double preciseValue = 3.14159;

    // Text and logic
    public string playerName = "Player";
    public bool isAlive = true;
    public char grade = 'A';

    // Unity types
    public Vector3 position;
    public Color playerColor = Color.red;

    void Start()
    {
        ShowVariableTypes();
        PerformCalculations();
    }

    void ShowVariableTypes()
    {
        Debug.Log("Health (int): " + playerHealth);
        Debug.Log("Speed (float): " + moveSpeed);
        Debug.Log("Name (string): " + playerName);
        Debug.Log("Alive (bool): " + isAlive);
    }

    void PerformCalculations()
    {
        int damage = 25;
        int newHealth = playerHealth - damage;
        Debug.Log($"Health after damage: {newHealth}");

        float distance = moveSpeed * Time.deltaTime;
        Debug.Log($"Distance per frame: {distance}");

        bool isHealthLow = playerHealth < 30;
        Debug.Log($"Low health? {isHealthLow}");
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.H))
        {
            playerHealth -= 10;
            Debug.Log("Damage! Health: " + playerHealth);
        }
    }
}

Example 2: Functions and Methods

Creating and using functions in C#:

using UnityEngine;

public class FunctionsExample : MonoBehaviour
{
    private int currentHealth = 100;
    private int experience = 0;

    void Start()
    {
        InitializePlayer();
        ShowPlayerInfo();
    }

    // Function without parameters
    void InitializePlayer()
    {
        currentHealth = 100;
        experience = 0;
        Debug.Log("Player initialized");
    }

    // Function with parameters
    void TakeDamage(int damage)
    {
        currentHealth -= damage;
        currentHealth = Mathf.Max(0, currentHealth);
        Debug.Log($"Damage: {damage}. Health: {currentHealth}");
    }

    // Function with return value
    bool IsAlive()
    {
        return currentHealth > 0;
    }

    // Function with multiple parameters
    float CalculateDamage(float baseDamage, float multiplier)
    {
        return baseDamage * multiplier;
    }

    // Function with default parameter
    void Heal(int healAmount = 20)
    {
        currentHealth += healAmount;
        Debug.Log($"Healing: {healAmount}. Health: {currentHealth}");
    }

    public void ShowPlayerInfo()
    {
        Debug.Log($"Health: {currentHealth}");
        Debug.Log($"Experience: {experience}");
        Debug.Log($"Alive: {IsAlive()}");
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.Alpha1))
        {
            float damage = CalculateDamage(20f, 1.5f);
            TakeDamage(Mathf.RoundToInt(damage));
        }

        if (Input.GetKeyDown(KeyCode.Alpha2))
        {
            Heal();
        }
    }
}

Example 3: Classes and Objects

Creating and using classes in C#:

using UnityEngine;

// Base character class
public class Character
{
    protected string characterName;
    protected int health;
    protected int maxHealth;

    public string Name { get { return characterName; } }
    public int Health { get { return health; } }
    public bool IsAlive => health > 0;

    public Character(string name, int maxHp)
    {
        characterName = name;
        maxHealth = maxHp;
        health = maxHealth;
    }

    public virtual void TakeDamage(int damage)
    {
        health -= damage;
        health = Mathf.Max(0, health);
        Debug.Log($"{characterName} took {damage} damage");
    }
}

// Player class
public class Player : Character
{
    private int experience;

    public Player(string name) : base(name, 100)
    {
        experience = 0;
    }

    public override void TakeDamage(int damage)
    {
        int reducedDamage = Mathf.Max(1, damage - 5);
        base.TakeDamage(reducedDamage);
        Debug.Log("Armor absorbed damage!");
    }

    public void AddExperience(int exp)
    {
        experience += exp;
        Debug.Log($"Experience gained: {exp}");
    }
}

// Enemy class
public class Enemy : Character
{
    private int attackPower;

    public Enemy(string name, int hp, int attack) : base(name, hp)
    {
        attackPower = attack;
    }

    public void Attack(Character target)
    {
        Debug.Log($"{characterName} attacks!");
        target.TakeDamage(attackPower);
    }
}

public class ClassesExample : MonoBehaviour
{
    private Player player;
    private Enemy enemy;

    void Start()
    {
        player = new Player("Hero");
        enemy = new Enemy("Goblin", 50, 15);
        
        Debug.Log("Characters created!");
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.Alpha1))
        {
            player.TakeDamage(10);
        }

        if (Input.GetKeyDown(KeyCode.Alpha2))
        {
            enemy.Attack(player);
        }
    }
}

Example 4: Arrays and Lists

Working with arrays and collections in C#:

using UnityEngine;
using System.Collections.Generic;

public class ArraysExample : MonoBehaviour
{
    // Arrays
    public string[] playerNames = {"Alex", "Maria", "John"};
    public int[] scores = {100, 250, 180};

    // Lists
    public List<string> inventory = new List<string>();

    void Start()
    {
        DemonstrateArrays();
        DemonstrateLists();
    }

    void DemonstrateArrays()
    {
        Debug.Log("=== Arrays ===");
        
        // Creating an array
        int[] numbers = new int[5];
        for (int i = 0; i < numbers.Length; i++)
        {
            numbers[i] = i * 10;
        }

        // Displaying array
        foreach (int number in numbers)
        {
            Debug.Log("Number: " + number);
        }

        // Working with names and scores
        for (int i = 0; i < playerNames.Length; i++)
        {
            Debug.Log($"{playerNames[i]} - {scores[i]} points");
        }
    }

    void DemonstrateLists()
    {
        Debug.Log("=== Lists ===");
        
        // Adding elements
        inventory.Add("Sword");
        inventory.Add("Shield");
        inventory.Add("Potion");

        // Displaying list
        for (int i = 0; i < inventory.Count; i++)
        {
            Debug.Log($"{i + 1}. {inventory[i]}");
        }

        // Searching and removing
        bool hasSword = inventory.Contains("Sword");
        Debug.Log("Has sword: " + hasSword);

        inventory.Remove("Potion");
        Debug.Log("Potion used");
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.Alpha1))
        {
            string[] items = {"Ring", "Amulet", "Scroll"};
            string randomItem = items[Random.Range(0, items.Length)];
            inventory.Add(randomItem);
            Debug.Log("Found: " + randomItem);
        }
    }
}

Example 5: Conditional Operators

Using conditions and loops in C#:

using UnityEngine;

public class ConditionsExample : MonoBehaviour
{
    public int playerLevel = 1;
    public int playerHealth = 100;
    public string playerClass = "Warrior";

    void Start()
    {
        DemonstrateConditions();
        DemonstrateLoops();
    }

    void DemonstrateConditions()
    {
        Debug.Log("=== Conditions ===");

        // Simple if condition
        if (playerHealth > 50)
        {
            Debug.Log("Health is OK");
        }

        // if-else
        if (playerLevel >= 10)
        {
            Debug.Log("Experienced player");
        }
        else
        {
            Debug.Log("Beginner");
        }

        // if-else if-else
        if (playerHealth > 80)
        {
            Debug.Log("Excellent health");
        }
        else if (playerHealth > 50)
        {
            Debug.Log("Good health");
        }
        else if (playerHealth > 20)
        {
            Debug.Log("Low health");
        }
        else
        {
            Debug.Log("Critical condition!");
        }

        // switch
        switch (playerClass)
        {
            case "Warrior":
                Debug.Log("Strong in close combat");
                break;
            case "Mage":
                Debug.Log("Masters magic");
                break;
            case "Archer":
                Debug.Log("Master of ranged combat");
                break;
            default:
                Debug.Log("Unknown class");
                break;
        }
    }

    void DemonstrateLoops()
    {
        Debug.Log("=== Loops ===");

        // for loop
        Debug.Log("For loop:");
        for (int i = 1; i <= 5; i++)
        {
            Debug.Log("Iteration: " + i);
        }

        // while loop
        Debug.Log("While loop:");
        int count = 0;
        while (count < 3)
        {
            Debug.Log("Count: " + count);
            count++;
        }

        // foreach loop with array
        string[] weapons = {"Sword", "Bow", "Staff"};
        Debug.Log("Weapons:");
        foreach (string weapon in weapons)
        {
            Debug.Log("- " + weapon);
        }
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.Alpha1))
        {
            playerHealth -= 20;
            CheckHealthStatus();
        }

        if (Input.GetKeyDown(KeyCode.Alpha2))
        {
            playerLevel++;
            CheckLevelRewards();
        }
    }

    void CheckHealthStatus()
    {
        if (playerHealth <= 0)
        {
            Debug.Log("Player died!");
        }
        else if (playerHealth < 30)
        {
            Debug.Log("Need healing!");
        }
    }

    void CheckLevelRewards()
    {
        if (playerLevel % 5 == 0)
        {
            Debug.Log("Level reward received!");
        }
    }
}
