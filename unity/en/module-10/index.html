Memory Management with Addressables

Efficient loading and unloading of resources for memory optimization.

using UnityEngine; using UnityEngine.AddressableAssets; using UnityEngine.ResourceManagement.AsyncOperations; using System.Collections.Generic; public class AddressableManager : MonoBehaviour { public AssetReference playerPrefab; public AssetReference audioClip; private Dictionary<string, AsyncOperationHandle> loadedAssets = new Dictionary<string, AsyncOperationHandle>(); void Start() { LoadAssets(); Debug.Log("Addressable Manager initialized"); } async void LoadAssets() { // Loading player prefab var playerHandle = playerPrefab.LoadAssetAsync<GameObject>(); await playerHandle.Task; if (playerHandle.Status == AsyncOperationStatus.Succeeded) { loadedAssets["player"] = playerHandle; var player = Instantiate(playerHandle.Result); Debug.Log("Player created from Addressable resource"); } // Loading audio var audioHandle = audioClip.LoadAssetAsync<AudioClip>(); await audioHandle.Task; if (audioHandle.Status == AsyncOperationStatus.Succeeded) { loadedAssets["audio"] = audioHandle; PlayAudio(audioHandle.Result); } } void PlayAudio(AudioClip clip) { var audioSource = GetComponent<AudioSource>(); if (audioSource == null) audioSource = gameObject.AddComponent<AudioSource>(); audioSource.clip = clip; audioSource.Play(); Debug.Log("Audio playing"); } public void UnloadAsset(string key) { if (loadedAssets.ContainsKey(key)) { Addressables.Release(loadedAssets[key]); loadedAssets.Remove(key); Debug.Log($"Resource {key} unloaded"); } } void OnDestroy() { foreach (var asset in loadedAssets.Values) { Addressables.Release(asset); } Debug.Log("All Addressable resources released"); } }

Cross-Platform Development

Adapting the game for different platforms: Android, WebGL, PC.

using UnityEngine; public class PlatformManager : MonoBehaviour { public bool enableTouchControls = false; public float mobileUIScale = 1.2f; public float desktopUIScale = 1.0f; void Start() { DetectAndConfigurePlatform(); } void DetectAndConfigurePlatform() { switch (Application.platform) { case RuntimePlatform.Android: case RuntimePlatform.IPhonePlayer: SetupMobile(); break; case RuntimePlatform.WebGLPlayer: SetupWebGL(); break; default: SetupDesktop(); break; } Debug.Log($"Platform: {Application.platform}"); } void SetupMobile() { enableTouchControls = true; SetUIScale(mobileUIScale); QualitySettings.SetQualityLevel(1); // Low quality // Enable touch elements var touchControls = GameObject.FindGameObjectsWithTag("TouchControl"); foreach (var control in touchControls) { control.SetActive(true); } Debug.Log("Mobile configuration applied"); } void SetupWebGL() { enableTouchControls = SystemInfo.deviceType == DeviceType.Handheld; SetUIScale(enableTouchControls ? mobileUIScale : desktopUIScale); QualitySettings.SetQualityLevel(2); // Medium quality Debug.Log("WebGL configuration applied"); } void SetupDesktop() { enableTouchControls = false; SetUIScale(desktopUIScale); QualitySettings.SetQualityLevel(4); // High quality // Disable touch elements var touchControls = GameObject.FindGameObjectsWithTag("TouchControl"); foreach (var control in touchControls) { control.SetActive(false); } Debug.Log("Desktop configuration applied"); } void SetUIScale(float scale) { var canvas = FindObjectOfType<Canvas>(); if (canvas != null) { var scaler = canvas.GetComponent<UnityEngine.UI.CanvasScaler>(); if (scaler != null) { scaler.scaleFactor = scale; } } } }

Analytics and Tracking

Collecting data about player behavior and game performance.

using UnityEngine; using System.Collections.Generic; public class AnalyticsManager : MonoBehaviour { public string gameVersion = "1.0.0"; private float sessionStartTime; private int levelCompletions = 0; void Start() { InitializeAnalytics(); TrackSessionStart(); } void InitializeAnalytics() { sessionStartTime = Time.time; Debug.Log("Analytics initialized"); } void TrackSessionStart() { var data = new Dictionary<string, object>(); data["device"] = SystemInfo.deviceModel; data["os"] = SystemInfo.operatingSystem; data["version"] = gameVersion; data["platform"] = Application.platform.ToString(); TrackEvent("session_start", data); } public void TrackEvent(string eventName, Dictionary<string, object> parameters = null) { var eventData = new Dictionary<string, object>(); eventData["event"] = eventName; eventData["time"] = System.DateTime.Now.ToString(); eventData["session_time"] = Time.time - sessionStartTime; if (parameters != null) { foreach (var param in parameters) { eventData[param.Key] = param.Value; } } Debug.Log($"Event: {eventName}"); LogEvent(eventData); } public void TrackLevelComplete(int level, float time) { levelCompletions++; var data = new Dictionary<string, object>(); data["level"] = level; data["completion_time"] = time; data["total_completions"] = levelCompletions; TrackEvent("level_complete", data); } public void TrackPurchase(string item, float price) { var data = new Dictionary<string, object>(); data["item"] = item; data["price"] = price; TrackEvent("purchase", data); } void LogEvent(Dictionary<string, object> data) { string log = "Analytics:\n"; foreach (var item in data) { log += $" {item.Key}: {item.Value}\n"; } Debug.Log(log); } void OnDestroy() { var sessionTime = Time.time - sessionStartTime; var data = new Dictionary<string, object>(); data["duration"] = sessionTime; data["completions"] = levelCompletions; TrackEvent("session_end", data); } }

Monetization with Unity Ads

Integrating ads and in-app purchases for monetization.

using UnityEngine; using UnityEngine.UI; public class MonetizationManager : MonoBehaviour { [Header("Settings")] public string gameId = "1234567"; public Button watchAdButton; public Text coinsText; private int playerCoins = 0; private bool adReady = false; void Start() { InitializeMonetization(); LoadPlayerData(); } void InitializeMonetization() { // Simulating Unity Ads initialization Invoke("SetAdReady", 2f); if (watchAdButton != null) { watchAdButton.onClick.AddListener(ShowRewardedAd); } Debug.Log("Monetization initialized"); } void SetAdReady() { adReady = true; UpdateUI(); Debug.Log("Ad ready"); } void LoadPlayerData() { playerCoins = PlayerPrefs.GetInt("Coins", 0); UpdateUI(); } void SavePlayerData() { PlayerPrefs.SetInt("Coins", playerCoins); PlayerPrefs.Save(); } public void ShowRewardedAd() { if (!adReady) return; Debug.Log("Showing ad..."); adReady = false; UpdateUI(); // Simulating ad watching Invoke("OnAdComplete", 3f); } void OnAdComplete() { int reward = 50; playerCoins += reward; SavePlayerData(); UpdateUI(); Debug.Log($"Received {reward} coins for ad"); // Next ad in 30 seconds Invoke("SetAdReady", 30f); } public void PurchaseCoins(int amount, float price) { playerCoins += amount; SavePlayerData(); UpdateUI(); // Send event to analytics var analytics = FindObjectOfType<AnalyticsManager>(); if (analytics != null) { analytics.TrackPurchase($"coins_{amount}", price); } Debug.Log($"Purchased {amount} coins for ${price}"); } public bool SpendCoins(int amount) { if (playerCoins >= amount) { playerCoins -= amount; SavePlayerData(); UpdateUI(); Debug.Log($"Spent {amount} coins"); return true; } Debug.Log("Insufficient coins"); return false; } void UpdateUI() { if (coinsText != null) { coinsText.text = $"Coins: {playerCoins}"; } if (watchAdButton != null) { watchAdButton.interactable = adReady; } } }

Preparing for Publishing

Automating the build process and preparing the game for release.

public string version = "1.0.0"; #if UNITY_EDITOR [ContextMenu("Build Android")] public void BuildAndroid() { SetupBuildSettings(); var buildOptions = new BuildPlayerOptions(); buildOptions.scenes = GetScenePaths(); buildOptions.locationPathName = $"Builds/Android/{productName}.apk"; buildOptions.target = BuildTarget.Android; buildOptions.options = BuildOptions.None; Debug.Log("Starting Android build..."); var report = BuildPipeline.BuildPlayer(buildOptions); if (report.summary.result == BuildResult.Succeeded) { Debug.Log("Android build completed successfully!"); } else { Debug.LogError("Android build failed"); } } [ContextMenu("Build Windows")] public void BuildWindows() { SetupBuildSettings(); var buildOptions = new BuildPlayerOptions(); buildOptions.scenes = GetScenePaths(); buildOptions.locationPathName = $"Builds/Windows/{productName}.exe"; buildOptions.target = BuildTarget.StandaloneWindows64; buildOptions.options = BuildOptions.None; Debug.Log("Starting Windows build..."); var report = BuildPipeline.BuildPlayer(buildOptions); if (report.summary.result == BuildResult.Succeeded) { Debug.Log("Windows build completed successfully!"); } } void SetupBuildSettings() { PlayerSettings.productName = productName; PlayerSettings.bundleVersion = version; // Android settings PlayerSettings.Android.bundleVersionCode++; PlayerSettings.Android.minSdkVersion = AndroidSdkVersions.AndroidApiLevel21; Debug.Log($"Build settings: {productName} v{version}"); } string[] GetScenePaths() { var scenes = new string[EditorBuildSettings.scenes.Length]; for (int i = 0; i < scenes.Length; i++) { scenes[i] = EditorBuildSettings.scenes[i].path; } return scenes; } [ContextMenu("Prepare for Store")] public void PrepareForStore() { // Setting store configurations PlayerSettings.Android.useAPKExpansionFiles = false; PlayerSettings.Android.keystoreName = "user.keystore"; Debug.Log("Game prepared for store publishing"); } #endif }
