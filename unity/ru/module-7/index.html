Основы работы с потоками

Введение в потоки и базовые операции с Thread классом.

using System;
using System.Threading;
using UnityEngine;

public class ThreadingBasics : MonoBehaviour
{
    [Header("Настройки потоков")]
    public int numberOfThreads = 4;
    public bool useBackgroundThreads = true;
    
    private Thread[] workerThreads;
    private bool isRunning = false;
    
    void Start()
    {
        InitializeThreads();
        Debug.Log("Система потоков инициализирована");
    }
    
    void InitializeThreads()
    {
        workerThreads = new Thread[numberOfThreads];
        
        for (int i = 0; i < numberOfThreads; i++)
        {
            int threadId = i;
            workerThreads[i] = new Thread(() => WorkerThreadMethod(threadId));
            workerThreads[i].IsBackground = useBackgroundThreads;
            workerThreads[i].Name = $"Рабочий поток {threadId}";
        }
    }
    
    void WorkerThreadMethod(int threadId)
    {
        Debug.Log($"Поток {threadId} запущен на ядре {Thread.CurrentThread.ManagedThreadId}");
        
        while (isRunning)
        {
            PerformWork(threadId);
            Thread.Sleep(1000);
        }
        
        Debug.Log($"Поток {threadId} завершен");
    }
    
    void PerformWork(int threadId)
    {
        int result = 0;
        for (int i = 0; i < 1000000; i++)
        {
            result += i % (threadId + 1);
        }
        
        Debug.Log($"Поток {threadId} завершил работу. Результат: {result}");
    }
    
    void Update()
    {
        if (Input.GetKeyDown(KeyCode.S))
        {
            StartThreads();
        }
        
        if (Input.GetKeyDown(KeyCode.T))
        {
            StopThreads();
        }
    }
    
    public void StartThreads()
    {
        if (!isRunning)
        {
            isRunning = true;
            
            for (int i = 0; i < workerThreads.Length; i++)
            {
                if (workerThreads[i].ThreadState == ThreadState.Unstarted)
                {
                    workerThreads[i].Start();
                }
            }
            
            Debug.Log("Все потоки запущены");
        }
    }
    
    public void StopThreads()
    {
        if (isRunning)
        {
            isRunning = false;
            Debug.Log("Остановка потоков...");
        }
    }
    
    void OnDestroy()
    {
        StopThreads();
    }
}

Task Parallel Library

Использование современного API для параллельного программирования.

using System;
using System.Collections.Generic;
using System.Threading.Tasks;
using UnityEngine;

public class TaskParallelController : MonoBehaviour
{
    [Header("Настройки задач")]
    public int dataSize = 1000000;
    
    private List<int> largeDataSet;
    private CancellationTokenSource cancellationTokenSource;
    
    void Start()
    {
        InitializeData();
        Debug.Log("Task Parallel Library Controller инициализирован");
    }
    
    void InitializeData()
    {
        largeDataSet = new List<int>(dataSize);
        for (int i = 0; i < dataSize; i++)
        {
            largeDataSet.Add(UnityEngine.Random.Range(1, 1000));
        }
        
        Debug.Log($"Создан набор данных размером {dataSize} элементов");
    }
    
    void Update()
    {
        if (Input.GetKeyDown(KeyCode.P))
        {
            ProcessDataParallel();
        }
        
        if (Input.GetKeyDown(KeyCode.S))
        {
            ProcessDataSequential();
        }
        
        if (Input.GetKeyDown(KeyCode.A))
        {
            ProcessDataAsync();
        }
    }
    
    public void ProcessDataParallel()
    {
        Debug.Log("Начало параллельной обработки...");
        var stopwatch = System.Diagnostics.Stopwatch.StartNew();
        
        long sum = 0;
        object lockObject = new object();
        
        Parallel.ForEach(largeDataSet, (number) =>
        {
            long result = ComputeExpensiveOperation(number);
            
            lock (lockObject)
            {
                sum += result;
            }
        });
        
        stopwatch.Stop();
        Debug.Log($"Параллельная обработка завершена за {stopwatch.ElapsedMilliseconds} мс. Сумма: {sum}");
    }
    
    public void ProcessDataSequential()
    {
        Debug.Log("Начало последовательной обработки...");
        var stopwatch = System.Diagnostics.Stopwatch.StartNew();
        
        long sum = 0;
        
        foreach (int number in largeDataSet)
        {
            sum += ComputeExpensiveOperation(number);
        }
        
        stopwatch.Stop();
        Debug.Log($"Последовательная обработка завершена за {stopwatch.ElapsedMilliseconds} мс. Сумма: {sum}");
    }
    
    public async void ProcessDataAsync()
    {
        Debug.Log("Начало асинхронной обработки...");
        cancellationTokenSource = new CancellationTokenSource();
        
        try
        {
            var result = await ProcessDataAsyncInternal(cancellationTokenSource.Token);
            Debug.Log($"Асинхронная обработка завершена. Результат: {result}");
        }
        catch (OperationCanceledException)
        {
            Debug.Log("Асинхронная операция была отменена");
        }
    }
    
    private async Task<long> ProcessDataAsyncInternal(CancellationToken cancellationToken)
    {
        return await Task.Run(() =>
        {
            long sum = 0;
            
            for (int i = 0; i < largeDataSet.Count; i++)
            {
                cancellationToken.ThrowIfCancellationRequested();
                sum += ComputeExpensiveOperation(largeDataSet[i]);
            }
            
            return sum;
        }, cancellationToken);
    }
    
    private long ComputeExpensiveOperation(int input)
    {
        long result = 0;
        for (int i = 0; i < input % 1000; i++)
        {
            result += (long)Math.Sqrt(i * input);
        }
        return result;
    }
}

Синхронизация потоков

Использование lock, Monitor и других примитивов синхронизации.

using System;
using System.Threading;
using System.Collections.Generic;
using UnityEngine;

public class ThreadSynchronization : MonoBehaviour
{
    [Header("Настройки синхронизации")]
    public int bufferSize = 10;
    
    private Queue<int> sharedBuffer;
    private readonly object bufferLock = new object();
    
    private volatile bool isRunning = false;
    private int itemsProduced = 0;
    private int itemsConsumed = 0;
    
    void Start()
    {
        InitializeSynchronization();
        Debug.Log("Система синхронизации потоков инициализирована");
    }
    
    void InitializeSynchronization()
    {
        sharedBuffer = new Queue<int>(bufferSize);
    }
    
    void Update()
    {
        if (Input.GetKeyDown(KeyCode.S))
        {
            StartProducerConsumer();
        }
        
        if (Input.GetKeyDown(KeyCode.T))
        {
            StopProducerConsumer();
        }
        
        if (Input.GetKeyDown(KeyCode.I))
        {
            ShowSyncInfo();
        }
    }
    
    public void StartProducerConsumer()
    {
        if (!isRunning)
        {
            isRunning = true;
            itemsProduced = 0;
            itemsConsumed = 0;
            
            Thread producer = new Thread(ProducerMethod);
            Thread consumer = new Thread(ConsumerMethod);
            
            producer.IsBackground = true;
            consumer.IsBackground = true;
            
            producer.Start();
            consumer.Start();
            
            Debug.Log("Производитель и потребитель запущены");
        }
    }
    
    void ProducerMethod()
    {
        while (isRunning)
        {
            lock (bufferLock)
            {
                if (sharedBuffer.Count < bufferSize)
                {
                    int item = UnityEngine.Random.Range(1, 1000);
                    sharedBuffer.Enqueue(item);
                    Interlocked.Increment(ref itemsProduced);
                    
                    Debug.Log($"Произведен элемент {item}. Буфер: {sharedBuffer.Count}");
                    Monitor.PulseAll(bufferLock);
                }
                else
                {
                    Monitor.Wait(bufferLock);
                }
            }
            
            Thread.Sleep(UnityEngine.Random.Range(100, 500));
        }
    }
    
    void ConsumerMethod()
    {
        while (isRunning)
        {
            lock (bufferLock)
            {
                while (sharedBuffer.Count == 0 && isRunning)
                {
                    Monitor.Wait(bufferLock);
                }
                
                if (sharedBuffer.Count > 0)
                {
                    int item = sharedBuffer.Dequeue();
                    Interlocked.Increment(ref itemsConsumed);
                    
                    Debug.Log($"Обработан элемент {item}. Буфер: {sharedBuffer.Count}");
                    Monitor.PulseAll(bufferLock);
                }
            }
            
            Thread.Sleep(UnityEngine.Random.Range(200, 800));
        }
    }
    
    public void StopProducerConsumer()
    {
        if (isRunning)
        {
            isRunning = false;
            
            lock (bufferLock)
            {
                Monitor.PulseAll(bufferLock);
            }
            
            Debug.Log("Остановка производителя и потребителя...");
        }
    }
    
    public void ShowSyncInfo()
    {
        lock (bufferLock)
        {
            Debug.Log($"Буфер: {sharedBuffer.Count}/{bufferSize}");
            Debug.Log($"Произведено: {itemsProduced}, Потреблено: {itemsConsumed}");
        }
    }
    
    void OnDestroy()
    {
        StopProducerConsumer();
    }
}

Многопоточность в Unity

Безопасная работа с потоками в Unity и взаимодействие с основным потоком.

using System;
using System.Collections;
using System.Collections.Concurrent;
using System.Threading.Tasks;
using UnityEngine;

public class UnityThreadingManager : MonoBehaviour
{
    [Header("Настройки Unity Threading")]
    public Transform[] objectsToMove;
    
    private ConcurrentQueue<Action> mainThreadActions = new ConcurrentQueue<Action>();
    private bool isProcessing = false;
    
    void Start()
    {
        Debug.Log("Unity Threading Manager инициализирован");
    }
    
    void Update()
    {
        ProcessMainThreadActions();
        
        if (Input.GetKeyDown(KeyCode.P))
        {
            StartPathfindingJobs();
        }
        
        if (Input.GetKeyDown(KeyCode.M))
        {
            StartBackgroundCalculations();
        }
        
        if (Input.GetKeyDown(KeyCode.L))
        {
            StartLongRunningTask();
        }
    }
    
    void ProcessMainThreadActions()
    {
        while (mainThreadActions.TryDequeue(out Action action))
        {
            try
            {
                action?.Invoke();
            }
            catch (Exception e)
            {
                Debug.LogError($"Ошибка выполнения действия: {e.Message}");
            }
        }
    }
    
    public void StartPathfindingJobs()
    {
        if (isProcessing) return;
        
        isProcessing = true;
        Debug.Log("Запуск фоновых задач поиска пути...");
        
        Task.Run(() =>
        {
            Vector3 calculatedPath = CalculatePathInBackground();
            
            mainThreadActions.Enqueue(() =>
            {
                ApplyPathResult(calculatedPath);
            });
        });
    }
    
    Vector3 CalculatePathInBackground()
    {
        System.Threading.Thread.Sleep(2000);
        
        Vector3 result = new Vector3(
            UnityEngine.Random.Range(-10f, 10f),
            0f,
            UnityEngine.Random.Range(-10f, 10f)
        );
        
        Debug.Log($"Фоновая задача завершена на потоке {System.Threading.Thread.CurrentThread.ManagedThreadId}");
        return result;
    }
    
    void ApplyPathResult(Vector3 newPath)
    {
        if (objectsToMove.Length > 0 && objectsToMove[0] != null)
        {
            StartCoroutine(MoveObjectToPosition(objectsToMove[0], newPath));
            Debug.Log($"Применен результат пути: {newPath}");
        }
        
        isProcessing = false;
    }
    
    IEnumerator MoveObjectToPosition(Transform obj, Vector3 targetPos)
    {
        Vector3 startPos = obj.position;
        float duration = 2f;
        float elapsed = 0f;
        
        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            float t = elapsed / duration;
            obj.position = Vector3.Lerp(startPos, targetPos, t);
            yield return null;
        }
        
        obj.position = targetPos;
    }
    
    public void StartBackgroundCalculations()
    {
        Debug.Log("Запуск фоновых вычислений...");
        
        Task.Run(async () =>
        {
            for (int i = 0; i < 10; i++)
            {
                await Task.Delay(500);
                
                int progress = (i + 1) * 10;
                
                mainThreadActions.Enqueue(() =>
                {
                    Debug.Log($"Прогресс: {progress}%");
                });
            }
            
            mainThreadActions.Enqueue(() =>
            {
                Debug.Log("Фоновые вычисления завершены!");
            });
        });
    }
    
    public async void StartLongRunningTask()
    {
        Debug.Log("Запуск долгосрочной задачи...");
        
        try
        {
            var result = await PerformLongRunningOperation();
            Debug.Log($"Долгосрочная задача завершена: {result}");
        }
        catch (Exception e)
        {
            Debug.LogError($"Ошибка в долгосрочной задаче: {e.Message}");
        }
    }
    
    private async Task<string> PerformLongRunningOperation()
    {
        return await Task.Run(() =>
        {
            System.Threading.Thread.Sleep(3000);
            return $"Операция завершена на потоке {System.Threading.Thread.CurrentThread.ManagedThreadId}";
        });
    }
    
    void OnDestroy()
    {
        isProcessing = false;
    }
}

Продвинутые техники многопоточности

Пулы потоков, CancellationToken и оптимизация производительности.

using System;
using System.Threading;
using System.Threading.Tasks;
using UnityEngine;

public class AdvancedThreadingController : MonoBehaviour
{
    [Header("Продвинутые настройки")]
    public int maxConcurrentTasks = 4;
    
    private SemaphoreSlim semaphore;
    private CancellationTokenSource cancellationTokenSource;
    
    void Start()
    {
        InitializeAdvancedThreading();
        Debug.Log("Advanced Threading Controller инициализирован");
    }
    
    void InitializeAdvancedThreading()
    {
        semaphore = new SemaphoreSlim(maxConcurrentTasks, maxConcurrentTasks);
        cancellationTokenSource = new CancellationTokenSource();
    }
    
    void Update()
    {
        if (Input.GetKeyDown(KeyCode.T))
        {
            StartTaskPool();
        }
        
        if (Input.GetKeyDown(KeyCode.C))
        {
            CancelAllTasks();
        }
        
        if (Input.GetKeyDown(KeyCode.R))
        {
            StartReaderWriterExample();
        }
    }
    
    public async void StartTaskPool()
    {
        Debug.Log("Запуск пула задач...");
        
        var tasks = new Task[10];
        
        for (int i = 0; i < tasks.Length; i++)
        {
            int taskId = i;
            tasks[i] = ProcessTaskWithSemaphore(taskId, cancellationTokenSource.Token);
        }
        
        try
        {
            await Task.WhenAll(tasks);
            Debug.Log("Все задачи в пуле завершены");
        }
        catch (OperationCanceledException)
        {
            Debug.Log("Задачи были отменены");
        }
    }
    
    private async Task ProcessTaskWithSemaphore(int taskId, CancellationToken cancellationToken)
    {
        await semaphore.WaitAsync(cancellationToken);
        
        try
        {
            Debug.Log($"Задача {taskId} начата");
            
            await Task.Delay(UnityEngine.Random.Range(1000, 3000), cancellationToken);
            
            Debug.Log($"Задача {taskId} завершена");
        }
        finally
        {
            semaphore.Release();
        }
    }
    
    public void CancelAllTasks()
    {
        cancellationTokenSource.Cancel();
        Debug.Log("Отмена всех задач...");
        
        cancellationTokenSource = new CancellationTokenSource();
    }
    
    private readonly ReaderWriterLockSlim readerWriterLock = new ReaderWriterLockSlim();
    private int sharedResource = 0;
    
    public void StartReaderWriterExample()
    {
        Debug.Log("Запуск примера ReaderWriterLock...");
        
        // Запуск читателей
        for (int i = 0; i < 3; i++)
        {
            int readerId = i;
            Task.Run(() => ReaderTask(readerId));
        }
        
        // Запуск писателя
        Task.Run(() => WriterTask());
    }
    
    void ReaderTask(int readerId)
    {
        for (int i = 0; i < 5; i++)
        {
            readerWriterLock.EnterReadLock();
            try
            {
                Debug.Log($"Читатель {readerId} прочитал значение: {sharedResource}");
                Thread.Sleep(500);
            }
            finally
            {
                readerWriterLock.ExitReadLock();
            }
            
            Thread.Sleep(200);
        }
    }
    
    void WriterTask()
    {
        for (int i = 0; i < 3; i++)
        {
            readerWriterLock.EnterWriteLock();
            try
            {
                sharedResource++;
                Debug.Log($"Писатель обновил значение до: {sharedResource}");
                Thread.Sleep(1000);
            }
            finally
            {
                readerWriterLock.ExitWriteLock();
            }
            
            Thread.Sleep(500);
        }
    }
    
    void OnDestroy()
    {
        cancellationTokenSource?.Cancel();
        semaphore?.Dispose();
        readerWriterLock?.Dispose();
    }
}
