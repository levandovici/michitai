Модуль 5: Unity UI и Canvas

Комплексные примеры для системы пользовательского интерфейса Unity

Основы работы с Canvas

Управление Canvas и основные настройки UI системы.

using UnityEngine; using UnityEngine.UI; public class CanvasManager : MonoBehaviour { [Header("Canvas Settings")] public Canvas mainCanvas; public CanvasScaler canvasScaler; void Start() { if (mainCanvas == null) mainCanvas = GetComponent<Canvas>(); Debug.Log("Canvas Manager инициализирован"); SetupCanvas(); } void SetupCanvas() { mainCanvas.renderMode = RenderMode.ScreenSpaceOverlay; mainCanvas.sortingOrder = 0; if (canvasScaler != null) { canvasScaler.uiScaleMode = CanvasScaler.ScaleMode.ScaleWithScreenSize; canvasScaler.referenceResolution = new Vector2(1920, 1080); canvasScaler.matchWidthOrHeight = 0.5f; } Debug.Log("Canvas настроен для разрешения: 1920x1080"); } public void SetRenderMode(RenderMode mode) { mainCanvas.renderMode = mode; Debug.Log("Режим рендеринга изменен на: " + mode); } public void ToggleCanvas() { mainCanvas.enabled = !mainCanvas.enabled; Debug.Log("Canvas " + (mainCanvas.enabled ? "включен" : "выключен")); } }

Управление UI Элементами

Работа с основными UI компонентами: кнопки, текст, изображения, слайдеры.

using UnityEngine; using UnityEngine.UI; using TMPro; public class UIElementsController : MonoBehaviour { [Header("UI Elements")] public Button mainButton; public TextMeshProUGUI displayText; public Image statusImage; public Slider progressSlider; public Toggle settingsToggle; private int clickCounter = 0; void Start() { SetupUIElements(); Debug.Log("UI Elements Controller инициализирован"); } void SetupUIElements() { if (mainButton != null) { mainButton.onClick.AddListener(OnMainButtonClick); } if (progressSlider != null) { progressSlider.onValueChanged.AddListener(OnSliderValueChanged); progressSlider.minValue = 0f; progressSlider.maxValue = 100f; } if (settingsToggle != null) { settingsToggle.onValueChanged.AddListener(OnToggleChanged); } UpdateDisplay(); } public void OnMainButtonClick() { clickCounter++; Debug.Log("Кнопка нажата! Счетчик: " + clickCounter); if (progressSlider != null) { progressSlider.value = Mathf.Min(progressSlider.value + 10f, 100f); } UpdateDisplay(); } public void OnSliderValueChanged(float value) { Debug.Log("Значение слайдера: " + value.ToString("F1")); if (statusImage != null) { Color newColor = Color.Lerp(Color.red, Color.green, value / 100f); statusImage.color = newColor; } UpdateDisplay(); } public void OnToggleChanged(bool isOn) { Debug.Log("Переключатель: " + (isOn ? "Включен" : "Выключен")); if (mainButton != null) { mainButton.interactable = isOn; } UpdateDisplay(); } void UpdateDisplay() { if (displayText != null) { string info = $"Клики: {clickCounter}\n"; info += $"Прогресс: {(progressSlider != null ? progressSlider.value.ToString("F1") : "0")}%\n"; info += $"Настройки: {(settingsToggle != null ? (settingsToggle.isOn ? "ВКЛ" : "ВЫКЛ") : "НЕТ")}"; displayText.text = info; } } }

UI Взаимодействия и События

Обработка событий UI, анимации и интерактивные элементы.

using UnityEngine; using UnityEngine.UI; using UnityEngine.EventSystems; using TMPro; public class UIInteractionManager : MonoBehaviour, IPointerEnterHandler, IPointerExitHandler { [Header("Interactive Elements")] public Button[] menuButtons; public TextMeshProUGUI feedbackText; [Header("Animation Settings")] public float hoverScale = 1.1f; public float animationDuration = 0.3f; private Vector3 originalScale; void Start() { SetupInteractions(); originalScale = transform.localScale; Debug.Log("UI Interaction Manager инициализирован"); } void SetupInteractions() { for (int i = 0; i < menuButtons.Length; i++) { int buttonIndex = i; if (menuButtons[i] != null) { menuButtons[i].onClick.AddListener(() => OnMenuButtonClick(buttonIndex)); } } UpdateFeedback("Система взаимодействий готова"); } public void OnMenuButtonClick(int buttonIndex) { Debug.Log($"Нажата кнопка меню {buttonIndex}"); string[] buttonNames = {"Главная", "Настройки", "Профиль", "Выход"}; string buttonName = buttonIndex < buttonNames.Length ? buttonNames[buttonIndex] : "Неизвестная"; UpdateFeedback($"Выбрано: {buttonName}"); StartCoroutine(AnimateFeedback()); } public void OnPointerEnter(PointerEventData eventData) { Debug.Log("Курсор наведен на UI элемент"); UpdateFeedback("Курсор наведен"); StartCoroutine(ScaleTo(originalScale * hoverScale)); } public void OnPointerExit(PointerEventData eventData) { Debug.Log("Курсор покинул UI элемент"); UpdateFeedback("Курсор убран"); StartCoroutine(ScaleTo(originalScale)); } void UpdateFeedback(string message) { if (feedbackText != null) { feedbackText.text = $"[{System.DateTime.Now:HH:mm:ss}] {message}"; } } System.Collections.IEnumerator AnimateFeedback() { if (feedbackText != null) { Color originalColor = feedbackText.color; for (int i = 0; i < 3; i++) { feedbackText.color = Color.yellow; yield return new WaitForSeconds(0.1f); feedbackText.color = originalColor; yield return new WaitForSeconds(0.1f); } } } System.Collections.IEnumerator ScaleTo(Vector3 targetScale) { Vector3 startScale = transform.localScale; float elapsedTime = 0f; while (elapsedTime < animationDuration) { elapsedTime += Time.deltaTime; float t = elapsedTime / animationDuration; transform.localScale = Vector3.Lerp(startScale, targetScale, t); yield return null; } transform.localScale = targetScale; } }

Layout Groups и Адаптивная Верстка

Использование Layout Groups для автоматического размещения UI элементов.

using UnityEngine; using UnityEngine.UI; using TMPro; public class LayoutGroupManager : MonoBehaviour { [Header("Layout Groups")] public HorizontalLayoutGroup horizontalLayout; public VerticalLayoutGroup verticalLayout; public GridLayoutGroup gridLayout; [Header("Dynamic Content")] public GameObject buttonPrefab; public Transform dynamicContainer; public TextMeshProUGUI infoText; private int dynamicButtonCount = 0; void Start() { SetupLayoutGroups(); Debug.Log("Layout Group Manager инициализирован"); } void SetupLayoutGroups() { if (horizontalLayout != null) { horizontalLayout.spacing = 10f; horizontalLayout.childControlWidth = true; horizontalLayout.childControlHeight = true; horizontalLayout.childAlignment = TextAnchor.MiddleCenter; } if (verticalLayout != null) { verticalLayout.spacing = 10f; verticalLayout.childControlWidth = true; verticalLayout.childForceExpandWidth = true; verticalLayout.childAlignment = TextAnchor.UpperCenter; } if (gridLayout != null) { gridLayout.cellSize = new Vector2(100, 100); gridLayout.spacing = new Vector2(10, 10); gridLayout.constraint = GridLayoutGroup.Constraint.FixedColumnCount; gridLayout.constraintCount = 3; } UpdateInfo(); } public void AddDynamicButton() { if (buttonPrefab != null && dynamicContainer != null) { GameObject newButton = Instantiate(buttonPrefab, dynamicContainer); dynamicButtonCount++; Button buttonComponent = newButton.GetComponent<Button>(); if (buttonComponent != null) { int buttonIndex = dynamicButtonCount; buttonComponent.onClick.AddListener(() => OnDynamicButtonClick(buttonIndex)); TextMeshProUGUI buttonText = newButton.GetComponentInChildren<TextMeshProUGUI>(); if (buttonText != null) { buttonText.text = $"Кнопка {buttonIndex}"; } } Debug.Log($"Добавлена динамическая кнопка {dynamicButtonCount}"); UpdateInfo(); LayoutRebuilder.ForceRebuildLayoutImmediate(dynamicContainer.GetComponent<RectTransform>()); } } public void RemoveDynamicButton() { if (dynamicContainer != null && dynamicContainer.childCount > 0) { Transform lastChild = dynamicContainer.GetChild(dynamicContainer.childCount - 1); DestroyImmediate(lastChild.gameObject); dynamicButtonCount = Mathf.Max(0, dynamicButtonCount - 1); Debug.Log("Удалена последняя динамическая кнопка"); UpdateInfo(); LayoutRebuilder.ForceRebuildLayoutImmediate(dynamicContainer.GetComponent<RectTransform>()); } } public void OnDynamicButtonClick(int buttonIndex) { Debug.Log($"Нажата динамическая кнопка {buttonIndex}"); UpdateInfo($"Последнее действие: Кнопка {buttonIndex}"); } void UpdateInfo(string additionalInfo = "") { if (infoText != null) { string info = $"Динамических кнопок: {dynamicButtonCount}"; if (!string.IsNullOrEmpty(additionalInfo)) { info += $"\n{additionalInfo}"; } infoText.text = info; } } }

Продвинутые UI Техники

Анимации UI, маски, эффекты и оптимизация производительности.

using UnityEngine; using UnityEngine.UI; using TMPro; public class AdvancedUIController : MonoBehaviour { [Header("Animation Elements")] public RectTransform animatedPanel; public CanvasGroup fadeGroup; public Image progressBar; public TextMeshProUGUI animatedText; [Header("Animation Settings")] public float animationDuration = 1f; public AnimationCurve animationCurve = AnimationCurve.EaseInOut(0, 0, 1, 1); private bool isAnimating = false; void Start() { Debug.Log("Advanced UI Controller инициализирован"); } public void AnimatePanel() { if (!isAnimating && animatedPanel != null) { StartCoroutine(AnimatePanelCoroutine()); } } System.Collections.IEnumerator AnimatePanelCoroutine() { isAnimating = true; Vector3 startPos = animatedPanel.anchoredPosition; Vector3 targetPos = startPos + Vector3.right * 200f; float elapsedTime = 0f; while (elapsedTime < animationDuration) { elapsedTime += Time.deltaTime; float t = elapsedTime / animationDuration; float curveValue = animationCurve.Evaluate(t); animatedPanel.anchoredPosition = Vector3.Lerp(startPos, targetPos, curveValue); yield return null; } animatedPanel.anchoredPosition = targetPos; // Возвращаем обратно yield return new WaitForSeconds(0.5f); elapsedTime = 0f; while (elapsedTime < animationDuration) { elapsedTime += Time.deltaTime; float t = elapsedTime / animationDuration; float curveValue = animationCurve.Evaluate(t); animatedPanel.anchoredPosition = Vector3.Lerp(targetPos, startPos, curveValue); yield return null; } animatedPanel.anchoredPosition = startPos; isAnimating = false; Debug.Log("Анимация панели завершена"); } public void FadeInOut() { if (fadeGroup != null) { StartCoroutine(FadeCoroutine()); } } System.Collections.IEnumerator FadeCoroutine() { // Fade Out float startAlpha = fadeGroup.alpha; float elapsedTime = 0f; while (elapsedTime < animationDuration / 2) { elapsedTime += Time.deltaTime; float t = elapsedTime / (animationDuration / 2); fadeGroup.alpha = Mathf.Lerp(startAlpha, 0f, t); yield return null; } fadeGroup.alpha = 0f; // Fade In elapsedTime = 0f; while (elapsedTime < animationDuration / 2) { elapsedTime += Time.deltaTime; float t = elapsedTime / (animationDuration / 2); fadeGroup.alpha = Mathf.Lerp(0f, 1f, t); yield return null; } fadeGroup.alpha = 1f; Debug.Log("Fade анимация завершена"); } public void AnimateProgressBar() { if (progressBar != null) { StartCoroutine(ProgressBarCoroutine()); } } System.Collections.IEnumerator ProgressBarCoroutine() { float elapsedTime = 0f; while (elapsedTime < animationDuration * 2) { elapsedTime += Time.deltaTime; float t = elapsedTime / (animationDuration * 2); progressBar.fillAmount = t; if (animatedText != null) { animatedText.text = $"Прогресс: {(t * 100):F0}%"; } yield return null; } progressBar.fillAmount = 1f; if (animatedText != null) { animatedText.text = "Прогресс: 100%"; } Debug.Log("Анимация прогресс-бара завершена"); } public void TypewriterEffect(string text) { if (animatedText != null) { StartCoroutine(TypewriterCoroutine(text)); } } System.Collections.IEnumerator TypewriterCoroutine(string text) { animatedText.text = ""; for (int i = 0; i <= text.Length; i++) { animatedText.text = text.Substring(0, i); yield return new WaitForSeconds(0.05f); } Debug.Log("Эффект печатной машинки завершен"); } void Update() { if (Input.GetKeyDown(KeyCode.Alpha1)) { AnimatePanel(); } if (Input.GetKeyDown(KeyCode.Alpha2)) { FadeInOut(); } if (Input.GetKeyDown(KeyCode.Alpha3)) { AnimateProgressBar(); } if (Input.GetKeyDown(KeyCode.Alpha4)) { TypewriterEffect("Это пример эффекта печатной машинки!"); } } }
