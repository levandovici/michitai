Пример: Подключение Web3 кошелька

Этот пример показывает, как подключить MetaMask кошелек к Unity игре:

using System;
using System.Threading.Tasks;
using UnityEngine;
using Nethereum.Web3;

/// 
/// Подключение к Web3 кошельку (MetaMask) из Unity
/// Демонстрирует установку соединения с блокчейн-кошельком
/// и выполнение базовых операций
/// 
public class WalletConnection : MonoBehaviour
{
    [Header("Конфигурация блокчейна")]
    [SerializeField] private string rpcUrl = "https://mainnet.infura.io/v3/YOUR_PROJECT_ID";
    
    private Web3 web3;
    public string PlayerWalletAddress { get; private set; }
    
    // События для обновления UI
    public event Action<string> OnWalletConnected;
    public event Action<decimal> OnBalanceUpdated;

    void Start()
    {
        InitializeWeb3();
    }

    /// 
    /// Инициализация соединения Web3
    /// 
    private void InitializeWeb3()
    {
        try
        {
            web3 = new Web3(rpcUrl);
            Debug.Log("Web3 соединение инициализировано успешно");
        }
        catch (Exception ex)
        {
            Debug.LogError($"Ошибка инициализации Web3: {ex.Message}");
        }
    }

    /// 
    /// Подключение к кошельку MetaMask
    /// 
    public async Task<bool> ConnectWallet()
    {
        try
        {
            // Запрос доступа к аккаунту
            var accounts = await RequestAccountAccess();
            
            if (accounts != null && accounts.Length > 0)
            {
                PlayerWalletAddress = accounts[0];
                Debug.Log($"Кошелек подключен: {PlayerWalletAddress}");
                
                OnWalletConnected?.Invoke(PlayerWalletAddress);
                await UpdateBalance();
                
                return true;
            }
            
            return false;
        }
        catch (Exception ex)
        {
            Debug.LogError($"Ошибка подключения кошелька: {ex.Message}");
            return false;
        }
    }

    /// 
    /// Получение текущего баланса ETH
    /// 
    public async Task UpdateBalance()
    {
        if (string.IsNullOrEmpty(PlayerWalletAddress)) return;

        try
        {
            var balanceWei = await web3.Eth.GetBalance.SendRequestAsync(PlayerWalletAddress);
            var balanceEth = Web3.Convert.FromWei(balanceWei.Value);
            
            Debug.Log($"Баланс кошелька: {balanceEth} ETH");
            OnBalanceUpdated?.Invoke(balanceEth);
        }
        catch (Exception ex)
        {
            Debug.LogError($"Ошибка получения баланса: {ex.Message}");
        }
    }
}

Пример: Управление NFT

Система для работы с NFT токенами в игре:

using System.Collections.Generic;
using UnityEngine;

/// 
/// Управление NFT в Unity игре
/// Загрузка, отображение и взаимодействие с NFT токенами
/// 
public class NFTManager : MonoBehaviour 
{
    [System.Serializable]
    public class GameNFT
    {
        public string tokenId;
        public string name;
        public string description;
        public string imageUrl;
        public string rarity;
        public Dictionary<string, object> attributes;
        
        public GameNFT(string id, string nftName, string desc, string image, string nftRarity)
        {
            tokenId = id;
            name = nftName;
            description = desc;
            imageUrl = image;
            rarity = nftRarity;
            attributes = new Dictionary<string, object>();
        }
    }

    private List<GameNFT> playerNFTs = new List<GameNFT>();
    
    /// 
    /// Загрузка NFT игрока из блокчейна
    /// 
    public async Task LoadPlayerNFTs(string playerAddress)
    {
        try
        {
            playerNFTs.Clear();
            
            // Получение количества NFT у игрока
            var balance = await GetNFTBalance(playerAddress);
            Debug.Log($"Игрок владеет {balance} NFT");
            
            // Загрузка каждого NFT
            for (int i = 0; i < balance; i++)
            {
                var tokenId = await GetTokenByIndex(playerAddress, i);
                var nft = await LoadNFTMetadata(tokenId);
                
                if (nft != null)
                {
                    playerNFTs.Add(nft);
                }
            }
            
            Debug.Log($"Загружено {playerNFTs.Count} NFT для игрока");
        }
        catch (Exception ex)
        {
            Debug.LogError($"Ошибка загрузки NFT: {ex.Message}");
        }
    }

    /// 
    /// Фильтрация NFT по редкости
    /// 
    public List<GameNFT> GetNFTsByRarity(string rarity)
    {
        return playerNFTs.FindAll(nft => 
            nft.rarity.Equals(rarity, StringComparison.OrdinalIgnoreCase));
    }
}

Пример: Смарт-контракт Solidity

Базовый смарт-контракт для игровых токенов:

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

/**
 * @title GameCoin
 * @dev Игровая валюта для Unity игры
 * Включает механики play-to-earn и внутриигровые покупки
 */
contract GameCoin is ERC20, Ownable {
    uint256 public constant MAX_SUPPLY = 1000000000 * 10**18; // 1 миллиард токенов
    
    // Маппинг для отслеживания достижений игроков
    mapping(address => uint256) public playerScore;
    mapping(address => uint256) public lastRewardClaim;
    
    // События для интеграции с игрой
    event RewardClaimed(address indexed player, uint256 amount, uint256 score);
    event ScoreUpdated(address indexed player, uint256 newScore);
    
    constructor() ERC20("GameCoin", "GC") {
        // Начальная эмиссия для владельца контракта
        _mint(msg.sender, 100000000 * 10**18); // 100 миллионов токенов
    }
    
    /**
     * @dev Обновление счета игрока (вызывается игровым сервером)
     * Только владелец контракта может обновлять счет
     */
    function updatePlayerScore(address player, uint256 score) external onlyOwner {
        playerScore[player] = score;
        emit ScoreUpdated(player, score);
    }
    
    /**
     * @dev Получение ежедневной награды игроком
     * Игроки могут получать награды раз в день на основе своего счета
     */
    function claimDailyReward() external {
        require(block.timestamp >= lastRewardClaim[msg.sender] + 1 days, 
                "Награда уже получена сегодня");
        require(playerScore[msg.sender] > 0, "Нет записанного счета");
        
        // Расчет награды на основе счета игрока (1 токен за 100 очков)
        uint256 reward = playerScore[msg.sender] / 100;
        require(reward > 0, "Счет слишком низкий для награды");
        
        // Проверка максимального предложения
        require(totalSupply() + reward <= MAX_SUPPLY, 
                "Превышение максимального предложения");
        
        // Обновление времени последнего получения награды
        lastRewardClaim[msg.sender] = block.timestamp;
        
        // Эмиссия токенов награды игроку
        _mint(msg.sender, reward);
        
        emit RewardClaimed(msg.sender, reward, playerScore[msg.sender]);
    }
    
    /**
     * @dev Сжигание токенов (для внутриигровых покупок)
     */
    function burn(uint256 amount) external {
        _burn(msg.sender, amount);
    }
}
