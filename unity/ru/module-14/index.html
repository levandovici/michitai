Итоговый проект

Создай свою полноценную игру, используя все изученные технологии, с опциональной интеграцией блокчейна

Этапы работы над проектом

1. Планирование проекта

Прежде чем приступать к кодингу, важно тщательно спланировать ваш проект:

  • Определите жанр и целевую аудиторию
  • Создайте концепт-документ (GDD)
  • Спланируйте архитектуру проекта
  • Определите ключевые механики и фичи
  • Создайте дорожную карту разработки
// Пример структуры проекта в Unity
Assets/
├── _Scripts/
│   ├── Core/         # Основные системы
│   ├── Gameplay/     # Игровые механики
│   ├── UI/           # Пользовательский интерфейс
│   └── Utils/        # Вспомогательные утилиты

└── Scenes/           # Игровые сцены

2. Разработка

Основной этап реализации вашей игры:

  • Настройте базовую сцену и камеру
  • Реализуйте основные игровые механики
  • Добавьте систему сохранений
  • Создайте пользовательский интерфейс
  • Добавьте звуки и визуальные эффекты
// Пример базового контроллера игрока
public class PlayerController : MonoBehaviour
{
    [Header("Movement")]
    public float moveSpeed = 5f;
    public float jumpForce = 10f;
    
    private Rigidbody rb;
    private bool isGrounded;

    void Start()
    {
        rb = GetComponent();
    }

    void Update()
    {
        // Обработка ввода и передвижение
        float moveX = Input.GetAxis("Horizontal");
        float moveZ = Input.GetAxis("Vertical");
        
        Vector3 movement = new Vector3(moveX, 0f, moveZ);
        rb.velocity = new Vector3(movement.x * moveSpeed, rb.velocity.y, movement.z * moveSpeed);

        // Прыжок
        if (Input.GetButtonDown("Jump") && isGrounded)
        {
            rb.AddForce(Vector3.up * jumpForce, ForceMode.Impulse);
            isGrounded = false;
        }
    }

    void OnCollisionEnter(Collision collision)
    {
        // Проверка на землю
        if (collision.gameObject.CompareTag("Ground"))
        {
            isGrounded = true;
        }
    }
}

3. Тестирование и отладка

Важные аспекты тестирования вашей игры:

  • Модульное тестирование отдельных компонентов
  • Интеграционное тестирование
  • Балансировка игрового процесса
  • Оптимизация производительности
  • Сбор и анализ отзывов
// Пример теста для игрового менеджера
using NUnit.Framework;
using UnityEngine;

public class GameManagerTests
{
    [Test]
    public void GameManager_AddScore_ScoreIncreases()
    {
        // Arrange
        var gameManager = new GameObject().AddComponent();
        int initialScore = gameManager.Score;
        int pointsToAdd = 100;

        // Act
        gameManager.AddScore(pointsToAdd);

        // Assert
        Assert.AreEqual(initialScore + pointsToAdd, gameManager.Score);
    }

    [Test]
    public void GameManager_GameOver_StateChangesToGameOver()
    {
        // Arrange
        var gameManager = new GameObject().AddComponent();
        
        // Act
        gameManager.GameOver();

        // Assert
        Assert.IsTrue(gameManager.IsGameOver);
    }
}

4. Документация

Подготовка проекта к сдаче:

  • Напишите README с инструкцией по запуску
  • Документируйте ключевые классы и методы
  • Подготовьте презентацию проекта
  • Создайте демо-видео
  • Опишите известные проблемы и планы по доработке
/// 
/// Управляет основным игровым процессом, включая счет, состояние игры и события.
/// 
public class GameManager : MonoBehaviour
{
    /// 
    /// Текущий счет игрока.
    /// 
    public int Score { get; private set; }

    /// 
    /// Возвращает true, если игра завершена.
    /// 
    public bool IsGameOver { get; private set; }

    /// 
    /// Добавляет очки к текущему счету.
    /// 
    /// Количество очков для добавления.
    public void AddScore(int points)
    {
        if (!IsGameOver)
        {
            Score += points;
            OnScoreUpdated?.Invoke(Score);
        }
    }

    /// 
    /// Завершает игру.
    /// 
    public void GameOver()
    {
        if (!IsGameOver)
        {
            IsGameOver = true;
            OnGameOver?.Invoke();
        }
    }

    // События
    public event System.Action OnScoreUpdated;
    public event System.Action OnGameOver;
}

Дополнительно: Интеграция с блокчейном

Для проектов с блокчейн-компонентами:

Пример 1: Интеграция Web3 кошелька

Подключение MetaMask и других Web3 кошельков к Unity игре:

using UnityEngine;
using System.Collections;

public class WalletManager : MonoBehaviour
{
    private string playerWalletAddress;
    private bool isWalletConnected = false;
    
    void Start()
    {
        Debug.Log("Web3 Wallet Manager инициализирован");
        ConnectWallet();
    }
    
    void ConnectWallet()
    {
        Debug.Log("Подключение к Web3 кошельку...");
        
        playerWalletAddress = "0x" + System.Guid.NewGuid().ToString("N").Substring(0, 40);
        isWalletConnected = true;
        
        Debug.Log($"Кошелек подключен: {playerWalletAddress}");
        GetWalletBalance();
    }
    
    void GetWalletBalance()
    {
        Debug.Log("Получение баланса кошелька...");
        
        float balance = Random.Range(0.1f, 5.0f);
        Debug.Log($"Баланс кошелька: {balance:F4} ETH");
    }
    
    public bool IsWalletConnected()
    {
        return isWalletConnected;
    }
}

Пример 2: NFT система в игре

Создание и управление NFT предметами в Unity:

using UnityEngine;
using System.Collections.Generic;

[System.Serializable]
public class NFTData
{
    public int tokenId;
    public string name;
    public string rarity;
    public int attack = 10;
    public int defense = 10;
}

public class NFTManager : MonoBehaviour
{
    private List<NFTData> playerNFTs = new List<NFTData>();
    
    void Start()
    {
        CreateSampleNFTs();
        Debug.Log("NFT Manager инициализирован");
    }
    
    void CreateSampleNFTs()
    {
        playerNFTs.Add(new NFTData
        {
            tokenId = 1,
            name = "Огненный меч",
            rarity = "Редкий",
            attack = 25,
            defense = 5
        });
        
        playerNFTs.Add(new NFTData
        {
            tokenId = 2,
            name = "Ледяной щит",
            rarity = "Эпический",
            attack = 5,
            defense = 30
        });
        
        Debug.Log($"Создано {playerNFTs.Count} NFT предметов");
    }
    
    public NFTData GetNFTById(int tokenId)
    {
        return playerNFTs.Find(nft => nft.tokenId == tokenId);
    }
    
    public void TransferNFT(int tokenId, string toAddress)
    {
        var nft = GetNFTById(tokenId);
        if (nft != null)
        {
            Debug.Log($"NFT {nft.name} передан на адрес: {toAddress}");
        }
    }
}

Пример 3: Взаимодействие со смарт-контрактами

Вызов функций смарт-контрактов для игровой логики:

using UnityEngine;
using System.Collections;

public class SmartContractManager : MonoBehaviour
{
    private string contractAddress = "0x742d35Cc6634C0532925a3b8D4C2Aa8Db7C4E3C8";
    private string playerAddress;
    
    void Start()
    {
        playerAddress = FindObjectOfType<WalletManager>()?.playerWalletAddress ?? "0x123...456";
        Debug.Log("Smart Contract Manager инициализирован");
    }
    
    public void CallMintFunction(int tokenId, string tokenURI)
    {
        Debug.Log($"Вызов функции mint({tokenId}, {tokenURI})");
        Debug.Log($"Контракт: {contractAddress}");
        Debug.Log($"От адреса: {playerAddress}");
        
        StartCoroutine(SimulateMintTransaction(tokenId));
    }
    
    IEnumerator SimulateMintTransaction(int tokenId)
    {
        Debug.Log("Отправка транзакции...");
        yield return new WaitForSeconds(2f);
        
        Debug.Log($"NFT #{tokenId} успешно создан!");
        Debug.Log("Транзакция подтверждена в блокчейне");
    }
    
    public void GetTokenBalance()
    {
        Debug.Log("Запрос баланса токенов...");
        int balance = Random.Range(100, 1000);
        Debug.Log($"Баланс игровых токенов: {balance} GAME");
    }
}

Пример 4: Токеномика игры

Система игровых токенов и экономика:

using UnityEngine;

public class TokenManager : MonoBehaviour
{
    [Header("Token Settings")]
    public string tokenName = "GameToken";
    public string tokenSymbol = "GAME";
    public int currentBalance = 1000;
    public int stakedAmount = 0;
    
    void Start()
    {
        Debug.Log($"Token Manager инициализирован. Баланс: {currentBalance} {tokenSymbol}");
    }
    
    public void AddTokens(int amount)
    {
        currentBalance += amount;
        Debug.Log($"Добавлено {amount} {tokenSymbol}. Новый баланс: {currentBalance}");
    }
    
    public bool SpendTokens(int amount)
    {
        if (currentBalance >= amount)
        {
            currentBalance -= amount;
            Debug.Log($"Потрачено {amount} {tokenSymbol}. Остаток: {currentBalance}");
            return true;
        }
        
        Debug.LogWarning("Недостаточно токенов");
        return false;
    }
    
    public void StakeTokens(int amount)
    {
        if (currentBalance >= amount)
        {
            currentBalance -= amount;
            stakedAmount += amount;
            Debug.Log($"Застейкано {amount} {tokenSymbol}");
        }
    }
    
    public int GetBalance()
    {
        return currentBalance;
    }
}

Пример 5: NFT маркетплейс

Торговая площадка для покупки и продажи NFT предметов:

using UnityEngine;
using System.Collections.Generic;

[System.Serializable]
public class MarketplaceItem
{
    public NFTData nft;
    public int price;
    public string seller;
}

public class MarketplaceManager : MonoBehaviour
{
    private List<MarketplaceItem> marketplaceItems = new List<MarketplaceItem>();
    
    void Start()
    {
        CreateSampleMarketplaceItems();
        Debug.Log("NFT Marketplace инициализирован");
    }
    
    void CreateSampleMarketplaceItems()
    {
        marketplaceItems.Add(new MarketplaceItem
        {
            nft = new NFTData
            {
                tokenId = 101,
                name = "Ледяной посох",
                rarity = "Эпический"
            },
            price = 500,
            seller = "0x123...456"
        });
        
        Debug.Log($"Создано {marketplaceItems.Count} предметов на маркетплейсе");
    }
    
    public void BuyItem(MarketplaceItem item)
    {
        var tokenManager = FindObjectOfType<TokenManager>();
        
        if (tokenManager != null && tokenManager.SpendTokens(item.price))
        {
            Debug.Log($"Куплен NFT: {item.nft.name} за {item.price} токенов");
        }
        else
        {
            Debug.LogWarning("Недостаточно токенов для покупки");
        }
    }
    
    public void SellItem(NFTData nft, int price)
    {
        var marketItem = new MarketplaceItem
        {
            nft = nft,
            price = price,
            seller = "Player"
        };
        
        marketplaceItems.Add(marketItem);
        Debug.Log($"NFT {nft.name} выставлен на продажу за {price} токенов");
    }
}

  4. Инструкцию по установке и запуску
